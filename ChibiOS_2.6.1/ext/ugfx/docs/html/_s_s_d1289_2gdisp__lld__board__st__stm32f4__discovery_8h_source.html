42 /**
43  * @brief Initialise the board for the display.
44  * @notes Performs the following functions:
45  * 1. initialise the io port used by your display
46  * 2. initialise the reset pin (initial state not-in-reset)
47  * 3. initialise the chip select pin (initial state not-active)
48  * 4. initialise the backlight pin (initial state back-light off)
49  *
50  * @notapi
51  */
52 static inline void init_board(void) {
53 
54  #if defined(STM32F1XX) || defined(STM32F3XX)
55  /* FSMC setup for F1/F3 */
56  rccEnableAHB(RCC_AHBENR_FSMCEN, 0);
57 
58  #if defined(GDISP_USE_DMA) && defined(GDISP_DMA_STREAM)
59  #error "DMA not implemented for F1/F3 Devices"
60  #endif
61  #elif defined(STM32F4XX) || defined(STM32F2XX)
62  /* STM32F2-F4 FSMC init */
63  rccEnableAHB3(RCC_AHB3ENR_FSMCEN, 0);
64 
65  #if defined(GDISP_USE_DMA) && defined(GDISP_DMA_STREAM)
66  if (dmaStreamAllocate(GDISP_DMA_STREAM, 0, NULL, NULL)) gfxExit();
67  dmaStreamSetMemory0(GDISP_DMA_STREAM, &GDISP_RAM);
68  dmaStreamSetMode(GDISP_DMA_STREAM, STM32_DMA_CR_PL(0) | STM32_DMA_CR_PSIZE_HWORD | STM32_DMA_CR_MSIZE_HWORD | STM32_DMA_CR_DIR_M2M);
69  #endif
70  #else
71  #error "FSMC not implemented for this device"
72  #endif
73 
74  /* set pins to FSMC mode */
75  IOBus busD = {GPIOD, (1 << 0) | (1 << 1) | (1 << 4) | (1 << 5) | (1 << 7) | (1 << 8) |
76  (1 << 9) | (1 << 10) | (1 << 11) | (1 << 14) | (1 << 15), 0};
77 
78  IOBus busE = {GPIOE, (1 << 7) | (1 << 8) | (1 << 9) | (1 << 10) | (1 << 11) | (1 << 12) |
79  (1 << 13) | (1 << 14) | (1 << 15), 0};
80 
81  palSetBusMode(&busD, PAL_MODE_ALTERNATE(12));
82  palSetBusMode(&busE, PAL_MODE_ALTERNATE(12));
83 
84  /* FSMC timing */
85  FSMC_Bank1->BTCR[FSMC_Bank+1] = FSMC_BTR1_ADDSET_0 | FSMC_BTR1_DATAST_2 | FSMC_BTR1_BUSTURN_0 ;
86 
87  /* Bank1 NOR/SRAM control register configuration
88  * This is actually not needed as already set by default after reset */
89  FSMC_Bank1->BTCR[FSMC_Bank] = FSMC_BCR1_MWID_0 | FSMC_BCR1_WREN | FSMC_BCR1_MBKEN;
90 
91  /* Display backlight control */
92  /* TIM3 is an alternate function 2 (AF2) */
93  pwmStart(&PWMD3, &pwmcfg);
94  palSetPadMode(GPIOB, 0, PAL_MODE_ALTERNATE(2));
95  pwmEnableChannel(&PWMD3, 2, 100);
96 
97 }
