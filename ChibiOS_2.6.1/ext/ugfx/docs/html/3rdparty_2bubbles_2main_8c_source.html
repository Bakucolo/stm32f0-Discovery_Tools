33 void initialize (void)
34 {
35  uint16_t i;

37  /* if you change the SCALE*1.25 back to SCALE, the program will
38  * occassionally overrun the cosi array -- however this actually
39  * produces some interesting effects as the BUBBLES LOSE CONTROL!!!!

41  for (i = 0; i < SCALE+(SCALE/4); i++)

47 void matrix (int16_t xyz[3][N], color_t col[N])
48 {
49  static uint32_t t = 0;
50  int16_t x = -SCALE, y = -SCALE;
51  uint16_t i, s, d;
52  uint8_t red,grn,blu;
53 
54 #define RED_COLORS (32)
55 #define GREEN_COLORS (64)
56 #define BLUE_COLORS (32)
57 
58  for (i = 0; i < N; i++)
59  {
60  xyz[0][i] = x;
61  xyz[1][i] = y;
62 
63  d = sqrtf(x * x + y * y); /* originally a fastsqrt() call */
64  s = sine[(t * 30) % SCALE] + SCALE;
65 
66  xyz[2][i] = sine[(d + s) % SCALE] * sine[(t * 10) % SCALE] / SCALE / 2;
67 
68  red = (cosi[xyz[2][i] + SCALE / 2] + SCALE) *
69  (RED_COLORS - 1) / SCALE / 2;
70  grn = (cosi[(xyz[2][i] + SCALE / 2 + 2 * SCALE / 3) % SCALE] + SCALE) *
71  (GREEN_COLORS - 1) / SCALE / 2;
72  blu = (cosi[(xyz[2][i] + SCALE / 2 + SCALE / 3) % SCALE] + SCALE) *
73  (BLUE_COLORS - 1) / SCALE / 2;
74  col[i] = ((red << 11) + (grn << 5) + blu);
75 
76  x += INCREMENT;
77 
78  if (x >= SCALE) x = -SCALE, y += INCREMENT;
79  }
80  t++;
81 }

84 void rotate (int16_t xyz[3][N], uint16_t angleX, uint16_t angleY, uint16_t angleZ)
85 {
86  uint16_t i;
87  int16_t tmpX, tmpY;
88  int16_t sinx = sine[angleX], cosx = cosi[angleX];
89  int16_t siny = sine[angleY], cosy = cosi[angleY];
90  int16_t sinz = sine[angleZ], cosz = cosi[angleZ];
91 
92  for (i = 0; i < N; i++)
93  {
94  tmpX = (xyz[0][i] * cosx - xyz[2][i] * sinx) / SCALE;
95  xyz[2][i] = (xyz[0][i] * sinx + xyz[2][i] * cosx) / SCALE;
96  xyz[0][i] = tmpX;
97 
98  tmpY = (xyz[1][i] * cosy - xyz[2][i] * siny) / SCALE;
99  xyz[2][i] = (xyz[1][i] * siny + xyz[2][i] * cosy) / SCALE;
100  xyz[1][i] = tmpY;
101 
102  tmpX = (xyz[0][i] * cosz - xyz[1][i] * sinz) / SCALE;
103  xyz[1][i] = (xyz[0][i] * sinz + xyz[1][i] * cosz) / SCALE;
104  xyz[0][i] = tmpX;
105  }
106 }

109 void draw(int16_t xyz[3][N], color_t col[N])
110 {
111  static uint16_t oldProjX[N] = {0};
112  static uint16_t oldProjY[N] = {0};
113  static uint8_t oldDotSize[N] = {0};
114  uint16_t i, projX, projY, projZ, dotSize;
115 
116  for (i = 0; i < N; i++)
117  {
118  projZ = SCALE - (xyz[2][i] + SCALE) / 4;
119  projX = width / 2 + (xyz[0][i] * projZ / SCALE) / 25;
120  projY = height / 2 + (xyz[1][i] * projZ / SCALE) / 25;
121  dotSize = 3 - (xyz[2][i] + SCALE) * 2 / SCALE;
122 
123  gdispDrawCircle (oldProjX[i], oldProjY[i], oldDotSize[i], background);
124 
125  if (projX > dotSize &&
126  projY > dotSize &&
127  projX < width - dotSize &&
128  projY < height - dotSize)
129  {
130  gdispDrawCircle (projX, projY, dotSize, col[i]);
131  oldProjX[i] = projX;
132  oldProjY[i] = projY;
133  oldDotSize[i] = dotSize;
134  }
135  }
136 }
