30 /**
31  * This demo demonstrates the use of the GAUDIN module to read audio channel 0.
32  * The audio channel gets read to display a very simple oscilloscope.
33  *
34  * It also demonstrates how to write your own custom GWIN window type.
35  */
36 #include "gfx.h"
37 
38 /* Include our custom gwin audio oscilloscope */
39 #include "gwinosc.h"
40 
41 /* Specify our timing parameters */
42 #define MY_AUDIO_FREQUENCY 4000 /* 4khz */
43 #define MY_AUDIO_CHANNEL 0 /* Use channel 0 */
44 
45 /* Data */
46 static GScopeObject gScopeWindow;
47 
48 /*
49  * Application entry point.
50  */
51 int main(void) {
52  GHandle ghScope;
53  coord_t swidth, sheight;
54 
55  gfxInit();
56 
57  /* Get the screen dimensions */
58  swidth = gdispGetWidth();
59  sheight = gdispGetHeight();
60 
61  /* Set up the scope window to fill the screen */
62  {
63  GWindowInit wi;
64 
65  wi.show = TRUE;
66  wi.x = wi.y = 0;
67  wi.width = swidth; wi.height = sheight;
68  ghScope = gwinScopeCreate(&gScopeWindow, &wi, MY_AUDIO_CHANNEL, MY_AUDIO_FREQUENCY);
69  }
70  gwinSetBgColor(ghScope, White);
71  gwinSetColor(ghScope, Red);
72  gwinClear(ghScope);
73 
74  /* Just keep displaying the scope traces */
75  while (TRUE) {
76  gwinScopeWaitForTrace(ghScope);
77  }
78 }
