62 static void ProcessEvent(void) {
63  XColor col;
64 
65  switch(evt.type) {
66  case Expose:
67  XCopyArea(dis, pix, win, gc,
68  evt.xexpose.x, evt.xexpose.y,
69  evt.xexpose.width, evt.xexpose.height,
70  evt.xexpose.x, evt.xexpose.y);
71  break;
72 #if GINPUT_NEED_MOUSE
73  case ButtonPress:
74  mousex = evt.xbutton.x;
75  mousey = evt.xbutton.y;
76  switch(evt.xbutton.button){
77  case 1: mousebuttons |= GINPUT_MOUSE_BTN_LEFT; break;
78  case 2: mousebuttons |= GINPUT_MOUSE_BTN_MIDDLE; break;
79  case 3: mousebuttons |= GINPUT_MOUSE_BTN_RIGHT; break;
80  case 4: mousebuttons |= GINPUT_MOUSE_BTN_4; break;
81  }
82  #if GINPUT_MOUSE_POLL_PERIOD == TIME_INFINITE
83  ginputMouseWakeup();
84  #endif
85  break;
86  case ButtonRelease:
87  mousex = evt.xbutton.x;
88  mousey = evt.xbutton.y;
89  switch(evt.xbutton.button){
90  case 1: mousebuttons &= ~GINPUT_MOUSE_BTN_LEFT; break;
91  case 2: mousebuttons &= ~GINPUT_MOUSE_BTN_MIDDLE; break;
92  case 3: mousebuttons &= ~GINPUT_MOUSE_BTN_RIGHT; break;
93  case 4: mousebuttons &= ~GINPUT_MOUSE_BTN_4; break;
94  }
95  #if GINPUT_MOUSE_POLL_PERIOD == TIME_INFINITE
96  ginputMouseWakeup();
97  #endif
98  break;
99  case MotionNotify:
100  mousex = evt.xmotion.x;
101  mousey = evt.xmotion.y;
102  #if GINPUT_MOUSE_POLL_PERIOD == TIME_INFINITE
103  ginputMouseWakeup();
104  #endif
105  break;
106 #endif
107  }
108 }

112 static DECLARE_THREAD_FUNCTION(ThreadX, arg) {
113  (void)arg;

133 bool_t gdisp_lld_init(void)
134 {
135  XSizeHints *pSH;
136  XSetWindowAttributes xa;
137  XTextProperty WindowTitle;
138  char * WindowTitleText;
139  gfxThreadHandle hth;
140 
141  #if GFX_USE_OS_LINUX || GFX_USE_OS_OSX
142  XInitThreads();
143  #endif
144 
145  dis = XOpenDisplay(NULL);
146  scr = DefaultScreen(dis);
147 
148  #if GDISP_FORCE_24BIT
149  if (!XMatchVisualInfo(dis, scr, 24, TrueColor, &vis)) {
150  fprintf(stderr, "Your display has no TrueColor mode\n");
151  XCloseDisplay(dis);
152  return FALSE;
153  }
154  cmap = XCreateColormap(dis, RootWindow(dis, scr),
155  vis.visual, AllocNone);
156  #else
157  vis.visual = CopyFromParent;
158  vis.depth = DefaultDepth(dis, scr);
159  cmap = DefaultColormap(dis, scr);
160  #endif
161  fprintf(stderr, "Running GFX Window in %d bit color\n", vis.depth);
162 
163  xa.colormap = cmap;
164  xa.border_pixel = 0xFFFFFF;
165  xa.background_pixel = 0x000000;
166 
167  win = XCreateWindow(dis, RootWindow(dis, scr), 16, 16,
168  GDISP_SCREEN_WIDTH, GDISP_SCREEN_HEIGHT,
169  0, vis.depth, InputOutput, vis.visual,
170  CWBackPixel|CWColormap|CWBorderPixel, &xa);
171  XSync(dis, TRUE);
172 
173  WindowTitleText = "GFX";
174  XStringListToTextProperty(&WindowTitleText, 1, &WindowTitle);
175  XSetWMName(dis, win, &WindowTitle);
176  XSetWMIconName(dis, win, &WindowTitle);
177  XSync(dis, TRUE);
178 
179  pSH = XAllocSizeHints();
180  pSH->flags = PSize | PMinSize | PMaxSize;
181  pSH->min_width = pSH->max_width = pSH->base_width = GDISP_SCREEN_WIDTH;
182  pSH->min_height = pSH->max_height = pSH->base_height = GDISP_SCREEN_HEIGHT;
183  XSetWMNormalHints(dis, win, pSH);
184  XFree(pSH);
185  XSync(dis, TRUE);
186 
187  pix = XCreatePixmap(dis, win,
188  GDISP_SCREEN_WIDTH, GDISP_SCREEN_HEIGHT, vis.depth);
189  XSync(dis, TRUE);
190 
191  gc = XCreateGC(dis, win, 0, 0);
192  XSetBackground(dis, gc, BlackPixel(dis, scr));
193  XSync(dis, TRUE);
194 
195  XSelectInput(dis, win, StructureNotifyMask);
196  XMapWindow(dis, win);
197  do { XNextEvent(dis, &evt); } while (evt.type != MapNotify);
198 
199  /* start the X11 thread */
200  XSetIOErrorHandler(FatalXIOError);
201  XSelectInput(dis, win,
202  ExposureMask | ButtonPressMask | ButtonReleaseMask | PointerMotionMask);
203 
204  if (!(hth = gfxThreadCreate(waXThread, sizeof(waXThread), HIGH_PRIORITY, ThreadX, 0))) {
205  fprintf(stderr, "Cannot start X Thread\n");
206  XCloseDisplay(dis);
207  exit(0);
208  }
209  #if GFX_USE_OS_LINUX || GFX_USE_OS_OSX
210  pthread_detach(hth);
211  #endif
212  gfxThreadClose(hth);
213 
214  /* Initialise the GDISP structure to match */
215  GDISP.Orientation = GDISP_ROTATE_0;
216  GDISP.Powermode = powerOn;
217  GDISP.Backlight = 100;
218  GDISP.Contrast = 50;
219  GDISP.Width = GDISP_SCREEN_WIDTH;
220  GDISP.Height = GDISP_SCREEN_HEIGHT;
221  #if GDISP_NEED_VALIDATION || GDISP_NEED_CLIP
222  GDISP.clipx0 = 0;
223  GDISP.clipy0 = 0;
224  GDISP.clipx1 = GDISP.Width;
225  GDISP.clipy1 = GDISP.Height;
226  #endif
227  return TRUE;
228 }

230 void gdisp_lld_draw_pixel(coord_t x, coord_t y, color_t color)
231 {
232  XColor col;
233 
234  #if GDISP_NEED_VALIDATION || GDISP_NEED_CLIP
235  // Clip pre orientation change
236  if (x < GDISP.clipx0 || y < GDISP.clipy0 || x >= GDISP.clipx1 || y >= GDISP.clipy1) return;
237  #endif
238 
239  col.red = RED_OF(color) << 8;
240  col.green = GREEN_OF(color) << 8;
241  col.blue = BLUE_OF(color) << 8;
242  XAllocColor(dis, cmap, &col);
243  XSetForeground(dis, gc, col.pixel);
244  XDrawPoint(dis, pix, gc, (int)x, (int)y );
245  XDrawPoint(dis, win, gc, (int)x, (int)y );
246  XFlush(dis);
247 }

249 void gdisp_lld_fill_area(coord_t x, coord_t y, coord_t cx, coord_t cy, color_t color) {
250  XColor col;
251 
252  #if GDISP_NEED_VALIDATION || GDISP_NEED_CLIP
253  // Clip pre orientation change
254  if (x < GDISP.clipx0) { cx -= GDISP.clipx0 - x; x = GDISP.clipx0; }
255  if (y < GDISP.clipy0) { cy -= GDISP.clipy0 - y; y = GDISP.clipy0; }
256  if (cx <= 0 || cy <= 0 || x >= GDISP.clipx1 || y >= GDISP.clipy1) return;
257  if (x+cx > GDISP.clipx1) cx = GDISP.clipx1 - x;
258  if (y+cy > GDISP.clipy1) cy = GDISP.clipy1 - y;
259  #endif
260 
261  col.red = RED_OF(color) << 8;
262  col.green = GREEN_OF(color) << 8;
263  col.blue = BLUE_OF(color) << 8;
264  XAllocColor(dis, cmap, &col);
265  XSetForeground(dis, gc, col.pixel);
266  XFillRectangle(dis, pix, gc, x, y, cx, cy);
267  XFillRectangle(dis, win, gc, x, y, cx, cy);
268  XFlush(dis);
269 }
