51  init_board();

67  write_index(0x10); //SYSR bit[4:3]=00 256 color bit[2:1]= 00 8bit MPU interface
68  write_data(0x0F); // if 16bit MCU interface and 65k color display
69 
70  write_index(0x04); //set PCLK invers
71  write_data(0x82);
72  gfxSleepMilliseconds(1);
73 
74  //Horizontal set
75  write_index(0x14); //HDWR//Horizontal Display Width Setting Bit[6:0]
76  write_data(0x3B);//Horizontal display width(pixels) = (HDWR + 1)*8
77  write_index(0x15); //Horizontal Non-Display Period Fine Tuning Option Register (HNDFTR)
78  write_data(0x00);//Horizontal Non-Display Period Fine Tuning(HNDFT) [3:0]
79  write_index(0x16); //HNDR//Horizontal Non-Display Period Bit[4:0]
80  write_data(0x01);//Horizontal Non-Display Period (pixels) = (HNDR + 1)*8
81  write_index(0x17); //HSTR//HSYNC Start Position[4:0]
82  write_data(0x00);//HSYNC Start Position(PCLK) = (HSTR + 1)*8
83  write_index(0x18); //HPWR//HSYNC Polarity ,The period width of HSYNC.
84  write_data(0x05);//HSYNC Width [4:0] HSYNC Pulse width(PCLK) = (HPWR + 1)*8
85 
86  //Vertical set
87  write_index(0x19); //VDHR0 //Vertical Display Height Bit [7:0]
88  write_data(0x0f); //Vertical pixels = VDHR + 1
89  write_index(0x1a); //VDHR1 //Vertical Display Height Bit [8]
90  write_data(0x01); //Vertical pixels = VDHR + 1
91  write_index(0x1b); //VNDR0 //Vertical Non-Display Period Bit [7:0]
92  write_data(0x02); //VSYNC Start Position(PCLK) = (VSTR + 1)
93  write_index(0x1c); //VNDR1 //Vertical Non-Display Period Bit [8]
94  write_data(0x00); //Vertical Non-Display area = (VNDR + 1)
95  write_index(0x1d); //VSTR0 //VSYNC Start Position[7:0]
96  write_data(0x07);//VSYNC Start Position(PCLK) = (VSTR + 1)
97  write_index(0x1e); //VSTR1 //VSYNC Start Position[8]
98  write_data(0x00);//VSYNC Start Position(PCLK) = (VSTR + 1)
99  write_index(0x1f); //VPWR //VSYNC Polarity ,VSYNC Pulse Width[6:0]
100  write_data(0x09); //VSYNC Pulse Width(PCLK) = (VPWR + 1)
101 
102  //Active window set
103  //setting active window X
104  write_index(0x30); //Horizontal Start Point 0 of Active Window (HSAW0)
105  write_data(0x00); //Horizontal Start Point of Active Window [7:0]
106  write_index(0x31); //Horizontal Start Point 1 of Active Window (HSAW1)
107  write_data(0x00); //Horizontal Start Point of Active Window [9:8]
108  write_index(0x34); //Horizontal End Point 0 of Active Window (HEAW0)
109  write_data(0xDF); //Horizontal End Point of Active Window [7:0]
110  write_index(0x35); //Horizontal End Point 1 of Active Window (HEAW1)
111  write_data(0x01); //Horizontal End Point of Active Window [9:8]
112 
113  //setting active window Y
114  write_index(0x32); //Vertical Start Point 0 of Active Window (VSAW0)
115  write_data(0x00); //Vertical Start Point of Active Window [7:0]
116  write_index(0x33); //Vertical Start Point 1 of Active Window (VSAW1)
117  write_data(0x00); //Vertical Start Point of Active Window [8]
118  write_index(0x36); //Vertical End Point of Active Window 0 (VEAW0)
119  write_data(0x0F); //Vertical End Point of Active Window [7:0]
120  write_index(0x37); //Vertical End Point of Active Window 1 (VEAW1)
121  write_data(0x01); //Vertical End Point of Active Window [8]
122 
123  // Display ON
124  write_index(0x01); //PWRR
125  write_data(0x80);
126 
127  // GPO0 DISP high
128  write_index(0x13); //GPO
129  write_data(0x01);
130 
131  set_backlight(0x80); //set to 90% brightness
132 
133  post_init_board();
134 
135  /* Initialise the GDISP structure to match */
136  GDISP.Width = GDISP_SCREEN_WIDTH;
137  GDISP.Height = GDISP_SCREEN_HEIGHT;
138  GDISP.Orientation = GDISP_ROTATE_0;
139  GDISP.Powermode = powerOn;
140  GDISP.Backlight = 100;
141  GDISP.Contrast = 50;
142  #if GDISP_NEED_VALIDATION || GDISP_NEED_CLIP
143  GDISP.clipx0 = 0;
144  GDISP.clipy0 = 0;
145  GDISP.clipx1 = GDISP.Width;
146  GDISP.clipy1 = GDISP.Height;
147  #endif
148 
149  return TRUE;
150 }

227  uint32_t area;
228 
229  #if GDISP_NEED_VALIDATION || GDISP_NEED_CLIP
230  if (x < GDISP.clipx0) { cx -= GDISP.clipx0 - x; x = GDISP.clipx0; }
231  if (y < GDISP.clipy0) { cy -= GDISP.clipy0 - y; y = GDISP.clipy0; }
232  if (cx <= 0 || cy <= 0 || x >= GDISP.clipx1 || y >= GDISP.clipy1) return;
233  if (x+cx > GDISP.clipx1) cx = GDISP.clipx1 - x;
234  if (y+cy > GDISP.clipy1) cy = GDISP.clipy1 - y;
235  #endif

237  area = cx*cy;
238 
239  gdisp_lld_setwindow(x, y, x+cx-1, y+cy-1);
240  write_index(RA8875_WRITE_MEMORY_START);
241 
242  #if defined(GDISP_USE_FSMC) && defined(GDISP_USE_DMA) && defined(GDISP_DMA_STREAM)
243  uint8_t i;
244  dmaStreamSetPeripheral(GDISP_DMA_STREAM, &color);
245  dmaStreamSetMode(GDISP_DMA_STREAM, STM32_DMA_CR_PL(0) | STM32_DMA_CR_PSIZE_HWORD | STM32_DMA_CR_MSIZE_HWORD | STM32_DMA_CR_DIR_M2M);

247  dmaStreamSetTransactionSize(GDISP_DMA_STREAM, 65535);
248  dmaStreamEnable(GDISP_DMA_STREAM);
249  dmaWaitCompletion(GDISP_DMA_STREAM);
250  }
251  dmaStreamSetTransactionSize(GDISP_DMA_STREAM, area%65535);
252  dmaStreamEnable(GDISP_DMA_STREAM);
253  dmaWaitCompletion(GDISP_DMA_STREAM);
