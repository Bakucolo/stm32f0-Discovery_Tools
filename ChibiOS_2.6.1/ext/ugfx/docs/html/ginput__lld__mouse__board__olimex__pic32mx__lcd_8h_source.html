60 /**
61  * @brief Check whether the surface is currently touched
62  * @return TRUE if the surface is currently touched
63  *
64  * @notapi
65  */
66 static inline bool_t getpin_pressed(void) {
67  adcsample_t samples[2] = {0, };
68 
69  // Set X+ to ground
70  palSetPadMode(IOPORTB, XPOS, PAL_MODE_OUTPUT);
71  palClearPad(IOPORTB, XPOS);
72 
73  // Set Y- to VCC
74  palSetPadMode(IOPORTB, YNEG, PAL_MODE_OUTPUT);
75  palSetPad(IOPORTB, YNEG);
76 
77  palSetPadMode(IOPORTB, XNEG, PAL_MODE_INPUT_ANALOG);
78  palSetPadMode(IOPORTB, YPOS, PAL_MODE_INPUT_ANALOG);
79 
80  adcConvert(&ADCD, &ADC_X_CG, &samples[0], 1);
81  adcConvert(&ADCD, &ADC_Y_CG, &samples[1], 1);
82 
83  return (ADC_MAX - (samples[1] - samples[0])) > TOUCH_THRESHOULD;
84 }
