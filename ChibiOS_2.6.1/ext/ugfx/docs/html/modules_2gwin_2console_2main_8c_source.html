32 /* The handles for our three consoles */
33 GHandle GW1, GW2, GW3;
34 
35 int main(void) {
36  uint8_t i;
37  font_t font1, font2;
38 
39  /* initialize and clear the display */
40  gfxInit();
41 
42  /* Set some fonts */
43  font1 = gdispOpenFont("UI2");
44  font2 = gdispOpenFont("DejaVu Sans 12");
45  gwinSetDefaultFont(font1);
46 
47  /* create the three console windows */
48  {
49  GWindowInit wi;
50 
51  wi.show = TRUE;
52  wi.x = 0; wi.y = 0; wi.width = gdispGetWidth(); wi.height = gdispGetHeight()/2;
53  GW1 = gwinConsoleCreate(NULL, &wi);
54  wi.y = gdispGetHeight()/2; wi.width = gdispGetWidth()/2; wi.height = gdispGetHeight();
55  GW2 = gwinConsoleCreate(NULL, &wi);
56  wi.x = gdispGetWidth()/2; wi.height = gdispGetHeight();
57  GW3 = gwinConsoleCreate(NULL, &wi);
58  }
59 
60  /* Use a special font for GW1 */
61  gwinSetFont(GW1, font2);
62 
63  /* Set the fore- and background colors for each console */
64  gwinSetColor(GW1, Green);
65  gwinSetBgColor(GW1, Black);
66  gwinSetColor(GW2, White);
67  gwinSetBgColor(GW2, Blue);
68  gwinSetColor(GW3, Black);
69  gwinSetBgColor(GW3, Red);
70 
71  /* clear all console windows - to set background */
72  gwinClear(GW1);
73  gwinClear(GW2);
74  gwinClear(GW3);
75 
76  /* Output some data on the first console */
77  for(i = 0; i < 10; i++) {
78  gwinPrintf(GW1, "Hello uGFX!\r\n");
79  }
80 
81  /* Output some data on the second console */
82  for(i = 0; i < 16; i++) {
83  gwinPrintf(GW2, "Message Nr.: %d\r\n", i+1);
84  }
85 
86  /* Output some data on the third console */
87  for(i = 0; i < 18; i++) {
88  gwinPrintf(GW3, "Message Nr.: %d\r\n", i+1);
89  }
90 
91  while(TRUE) {
92  gfxSleepMilliseconds(500);
93  }
94 }
