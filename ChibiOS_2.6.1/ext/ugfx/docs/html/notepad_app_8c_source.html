76 static void nbtnColorBarDraw(GHandle gh, bool_t enabled, bool_t isdown, const char *txt, const GButtonDrawStyle *pstyle, void *param) {
77  #define ccs nCurColorScheme
78 
79  int i, j, k;
80 
81  (void)txt;
82  (void)pstyle;
83  (void)param;
84  (void)enabled;
85 
86  ginputGetMouseStatus(0, &curPtr);
87 
88  // Draw the toolbars according to the mode
89  if (tbMode == 0) {
90  k = (curPtr.x - gh->x) / (NPAD_COLORBAR_WIDTH / 8);
91 
92  for (i = 0; i < 8; i++) {
93  j = gh->x + (NPAD_TOOLBAR_BTN_WIDTH / 2) + NPAD_TOOLBAR_BTN_WIDTH * i;
94 
95  if (isdown) {
96  // Update selection - this is like lazy release.
97  if (k >= 0 && k <= 7) {
98  selPenWidth = k + 1;
99  ncoreSetPenWidth((uint8_t) selPenWidth);
100  }
101 
102  gdispFillArea(gh->x + NPAD_TOOLBAR_BTN_WIDTH * i, gh->y,
103  NPAD_TOOLBAR_BTN_WIDTH, NPAD_TOOLBAR_BTN_HEIGHT,
104  selPenWidth - i == 1 ? ccs.toolbarBgActive : ccs.toolbarBgUnsel);
105  }
106  else {
107  gdispFillArea(gh->x + NPAD_TOOLBAR_BTN_WIDTH * i, gh->y,
108  NPAD_TOOLBAR_BTN_WIDTH, NPAD_TOOLBAR_BTN_HEIGHT,
109  selPenWidth - i == 1 ? ccs.toolbarBgSel : ccs.toolbarBgUnsel);
110 
111  gdispDrawBox(gh->x + NPAD_TOOLBAR_BTN_WIDTH * i, gh->y,
112  NPAD_TOOLBAR_BTN_WIDTH, NPAD_TOOLBAR_BTN_HEIGHT,
113  selPenWidth - i == 1 ? ccs.toolbarSeparator: ccs.toolbarBgUnsel);
114  }
115 
116  gdispFillCircle(j, gh->y + 10, i + 1, myColors[selColorIndex]);
117  }
118 
119  } else {
120  k = (curPtr.x - gh->x) / (NPAD_COLORBAR_WIDTH / 8);
121 
122  for (i = 0; i < 8; i++) {
123  j = gh->x + (NPAD_TOOLBAR_BTN_WIDTH / 2) + NPAD_TOOLBAR_BTN_WIDTH * i;
124 
125  if (isdown) {
126  // Update selection - this is like lazy release.
127  if (k >= 0 && k <= 7) {
128  selColorIndex = k;
129  selColor = myColors[k];
130  ncoreSetPenColor(selColor);
131  }
132 
133  gdispFillArea(gh->x + NPAD_TOOLBAR_BTN_WIDTH * i, gh->y,
134  NPAD_TOOLBAR_BTN_WIDTH, NPAD_TOOLBAR_BTN_HEIGHT,
135  k == i ? ccs.toolbarBgActive : ccs.toolbarBgUnsel);
136  }
137  else {
138  gdispFillArea(gh->x + NPAD_TOOLBAR_BTN_WIDTH * i, gh->y,
139  NPAD_TOOLBAR_BTN_WIDTH, NPAD_TOOLBAR_BTN_HEIGHT,
140  selColorIndex == i ? ccs.toolbarBgSel : ccs.toolbarBgUnsel);
141 
142  gdispDrawBox(gh->x + NPAD_TOOLBAR_BTN_WIDTH * i, gh->y,
143  NPAD_TOOLBAR_BTN_WIDTH, NPAD_TOOLBAR_BTN_HEIGHT,
144  selColorIndex == i ? ccs.toolbarSeparator: ccs.toolbarBgUnsel);
145  }
146 
147  gdispFillCircle(j, gh->y + (NPAD_TOOLBAR_BTN_HEIGHT / 2), 3, myColors[i] );
148  }
149  }
150 
151 
152  #undef ccs
153 }

155 static void nbtnColorBarSelDraw(GHandle gh, bool_t enabled, bool_t isdown, const char *txt, const GButtonDrawStyle *pstyle, void *param) {
156 #define ccs nCurColorScheme
157 
158  int i, j = 0, k;
159  color_t ca, cb;
160  GEventMouse ptr;
161 
162  (void)txt;
163  (void)pstyle;
164  (void)param;
165  (void)enabled;
166 
167  // Get a copy of the pointer location
168  ginputGetMouseStatus(0, &ptr);
169 
170  // Get which button the pointer is on right now
171  k = (ptr.x - gh->x) / NPAD_TOOLBAR_BTN_WIDTH;
172 
173  gdispDrawBox(gh->x, gh->y, gh->width, gh->height, ccs.toolbarBgUnsel);
174  gdispDrawBox(gh->x + 1, gh->y + 1, gh->width - 2, gh->height - 2, ccs.toolbarBgUnsel);
175 
176  for (i = 0; i < 2; i++) {
177  if (isdown) {
178  // Update selection - this is like lazy release.
179  if (k == 0 || k == 1) {
180  tbMode = k;
181  j = 1;
182  }
183 
184  ca = (tbMode == i ? ccs.toolbarBgActive : ccs.toolbarBgUnsel);
185  }
186  else {
187  ca = (tbMode == i ? ccs.toolbarBgSel : ccs.toolbarBgUnsel);
188  }
189 
190  cb = (tbMode == i ? ccs.toolbarSeparator : ccs.toolbarBgUnsel);
191 
192  gdispFillArea(gh->x + NPAD_TOOLBAR_BTN_WIDTH * i,
193  gh->y,
194  NPAD_TOOLBAR_BTN_WIDTH,
195  NPAD_TOOLBAR_BTN_HEIGHT,
196  ca);
197 
198  gdispImageSetBgColor(&toolbarImageFilmstrip, ca);
199  gdispDrawBox(gh->x + NPAD_TOOLBAR_BTN_WIDTH * i, gh->y,
200  NPAD_TOOLBAR_BTN_WIDTH, NPAD_TOOLBAR_BTN_HEIGHT, cb);
201 
202  /* Draw both the icons */
203  gwinImageDraw(gh, &toolbarImageFilmstrip,
204  2 + NPAD_TOOLBAR_BTN_WIDTH * i,
205  2,
206  NPAD_ICON_WIDTH,
207  NPAD_ICON_HEIGHT,
208  NPAD_ICON_START(3 + i),
209  0);
210  }
211 
212  if (j)
213  gwinButtonDraw(H(btnColorBar));
214 
215  #undef ccs
216 }

314 static DECLARE_THREAD_STACK(waNotepadThread, NPAD_THD_WA_SIZE);
315 static DECLARE_THREAD_FUNCTION(notepadThread, param) {
316 
317  GEventMouse *pem;
318  GEventGWinButton *peb;
319  GHandle ghc;
320 
321  (void)param;
322 
323  /* Get the display dimensions */
324  swidth = gdispGetWidth();
325  sheight = gdispGetHeight();
326 
327  font = gdispOpenFont("UI2");
328 
329  /* Initialize the mouse */
330  geventListenerInit(&gl);
331  ginputGetMouse(0);
332 
333  initButtons();
334 
335  /* Configure the GIF decoder with the toolbar Icon images */
336  gdispImageSetMemoryReader(&toolbarImageFilmstrip, toolbarIcons);
337  gdispImageOpen(&toolbarImageFilmstrip);
338 
339  /* Set clip to the entire screen */
340  gdispSetClip(0, 0, swidth, sheight);
341 
342  /* Clear the screen with the window background
343  * Also, draw the title bars */
344  gdispClear(nCurColorScheme.winBgColor);
345  gdispDrawBox(0, 0, swidth, sheight, nCurColorScheme.titleBarColor);
346  gdispFillArea(0, 0, swidth, NPAD_TITLEBAR_HEIGHT, nCurColorScheme.titleBarColor);
347  gdispDrawStringBox(NPAD_TITLETEXT_START_X,
348  NPAD_TITLETEXT_START_Y,
349  swidth,
350  NPAD_TITLEBAR_HEIGHT,
351  NPAD_TITLETEXT_STR,
352  font,
353  nCurColorScheme.titleTextColor,
354  justifyLeft);
355 
356  /* Create the drawing window, draw its border */
357  gdispDrawBox(NPAD_DRAWING_AREA_START_X - 1,
358  NPAD_DRAWING_AREA_START_Y - 1,
359  NPAD_DRAWING_AREA_WIDTH + 2,
360  NPAD_DRAWING_AREA_HEIGHT + 2,
361  nCurColorScheme.drawingWinBorder);
362 
363  nDrawingArea = gwinCreateWindow(NULL,
364  NPAD_DRAWING_AREA_START_X,
365  NPAD_DRAWING_AREA_START_Y,
366  NPAD_DRAWING_AREA_WIDTH,
367  NPAD_DRAWING_AREA_HEIGHT);
368 
369  /* Create the bottom status bar console */
370  ghc = gwinCreateConsole(NULL,
371  NPAD_STATUSBAR_START_X,
372  NPAD_STATUSBAR_START_Y,
373  NPAD_STATUSBAR_WIDTH,
374  NPAD_STATUSBAR_HEIGHT,
375  font);
376 
377  gdispImageDraw(&toolbarImageFilmstrip,
378  NPAD_STATUSBAR_ICON_START_X,
379  NPAD_STATUSBAR_ICON_START_Y,
380  NPAD_ICON_WIDTH,
381  NPAD_ICON_HEIGHT,
382  NPAD_ICON_START(12),
383  0);
384 
385  gwinSetBgColor(ghc, nCurColorScheme.winBgColor);
386  gwinSetColor(ghc, Black);
387 
388  gstatusConsole = ghc;
389 
390  /* draw the buttons */
391  gwinSetColor(nDrawingArea, Black);
392  gwinSetBgColor(nDrawingArea, White);
393 
394  gwinClear(nDrawingArea);
395  gwinClear(ghc);
396 
397  drawButtons();
398  drawVButtons();
399 
400  gwinPrintf(gstatusConsole, "Welcome to ChibiOS/GFX Notepad demo.");
401 
402  ncoreSpawnDrawThread(nDrawingArea, gstatusConsole);
403 
404  while(TRUE) {
405  pem = (GEventMouse *) geventEventWait(&gl, TIME_INFINITE);
406 
407  /* button pressed... */
408  if (pem->type == GEVENT_GWIN_BUTTON) {
409  peb = (GEventGWinButton *)pem;
410 
411  if (peb->button == H(btnNew)) {
412  // Reset all the settings
413  selColorIndex = 0;
414  selPenWidth = 0;
415  ncoreSetMode(NCORE_MODE_DRAW);
416 
417  gwinSetColor(nDrawingArea, Black);
418  gwinSetBgColor(nDrawingArea, White);
419 
420  // Refresh the buttons
421  drawButtons();
422  drawVButtons();
423 
424  gwinClear(nDrawingArea);
425  gwinPrintf(gstatusConsole, "\nScreen Cleared.");
426  }
427  else if (peb->button == H(btnOpen)) {
428  gwinPrintf(gstatusConsole, "\nFile Open not implemented.");
429  }
430  else if (peb->button == H(btnSave)) {
431  gwinPrintf(gstatusConsole, "\nFile Save not implemented.");
432  }
433  else if (peb->button == H(btnPencil)) {
434  ncoreSetMode(NCORE_MODE_DRAW);
435  drawVButtons();
436  gwinPrintf(gstatusConsole, "\nPencil Tool Selected.");
437  }
438  else if (peb->button == H(btnEraser)) {
439  ncoreSetMode(NCORE_MODE_ERASE);
440  drawVButtons();
441  gwinPrintf(gstatusConsole, "\nEraser Tool Selected.");
442  }
443  else if (peb->button == H(btnFill)) {
444  ncoreSetMode(NCORE_MODE_FILL);
445  drawVButtons();
446  gwinPrintf(gstatusConsole, "\nFill Tool Selected.");
447  }
448  else if (peb->button == H(btnClose)) {
449  break;
450  }
451  }
452  }
453 
454  gwinDestroyWindow(ghc);
455  // No need to destroy the buttons as they are statically allocated
456  gdispCloseFont(font);
457  ncoreTerminateDrawThread();
458  gdispImageClose(&toolbarImageFilmstrip);
459 
460  return 0;
461 }
