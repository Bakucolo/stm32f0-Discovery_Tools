33 /**
34  * --------------------------- Our Custom GWIN Oscilloscope ---------------
35  *
36  * This GWIN superset implements a simple audio oscilloscope using the GADC high speed device.
37  *
38  * It makes many assumptions, the most fundamental of which is that the audio device
39  * produces unsigned integer samples.
40  *
41  * The GMISC module with GMISC_NEED_ARRAYOPS could be used to process the samples more
42  * correctly if we were really building something generic.
43  */
44

48 #endif

50 /* Trigger methods */
51 #define TRIGGER_NONE 0 /* No triggering */
52 #define TRIGGER_POSITIVERAMP 1 /* Trigger on a positive going signal */
53 #define TRIGGER_MINVALUE 2 /* Trigger on reaching the minimum value from the last scope */

63 /* A scope window object. Treat it as a black box */
64 typedef struct GScopeObject_t {
65  GWindowObject g; // Base Class
66 
67  coord_t *lastscopetrace; // To store last scope trace
68  gfxSem bsem; // We get signalled on this
69  audin_sample_t *audiobuf; // To store audio samples
70  GEventAudioIn myEvent; // Information on received samples
71  coord_t nextx; // Where we are up to
72 #if TRIGGER_METHOD == TRIGGER_POSITIVERAMP
73  coord_t lasty; // The last y value - used for trigger slope detection
74 #elif TRIGGER_METHOD == TRIGGER_MINVALUE
75  coord_t lasty; // The last y value - used for trigger slope detection
76  coord_t scopemin; // The last scopes minimum value
77 #endif
78  } GScopeObject;
