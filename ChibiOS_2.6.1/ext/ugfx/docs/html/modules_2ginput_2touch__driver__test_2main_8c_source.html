32 static GConsoleObject gc;
33 static GButtonObject gNext;
34 static GButtonObject gPrev;
35 static GListener gl;

37 /*------------------------------------------------------------------------*
38  * GINPUT Touch Driver Calibrator. *
39  *------------------------------------------------------------------------*/
40 int main(void) {
41  GSourceHandle gs;
42  GEvent *pe;
43  GEventMouse *pem;
44  GEventGWinButton *peb;
45  coord_t swidth, sheight;
46  GHandle ghc, ghNext, ghPrev;
47  GEventType deviceType;
48  font_t font;
49  GWidgetInit wi;
50 
51  gfxInit(); // Initialize the display
52 
53  // Get the display dimensions
54  swidth = gdispGetWidth();
55  sheight = gdispGetHeight();
56  ghNext = ghPrev = 0;
57 
58  // Create our title
59  font = gdispOpenFont("UI2");
60  gwinSetDefaultFont(font);
61  gdispFillStringBox(0, 0, swidth, 20, "Touch Calibration", font, Red, White, justifyLeft);
62 
63  // Create our main display window
64  wi.g.show = TRUE; wi.g.x = 0; wi.g.y = 20; wi.g.width = swidth; wi.g.height = sheight-20;
65  ghc = gwinConsoleCreate(&gc, &wi.g);
66  gwinClear(ghc);
67 
68  // Initialize the mouse in our special no calibration mode.
69  geventListenerInit(&gl);
70  gs = ginputGetMouse(9999);
71  geventAttachSource(&gl, gs, GLISTEN_MOUSEDOWNMOVES|GLISTEN_MOUSEMETA);
72 
73  /*
74  * Test: Device Type
75  */
76 
77 StepDeviceType:
78  gwinClear(ghc);
79  gwinSetColor(ghc, Yellow);
80  gwinPrintf(ghc, "\n1. DEVICE TYPE\n\n");
81 
82  pem = (GEventMouse *)&gl.event;
83  ginputGetMouseStatus(0, pem);
84  deviceType = pem->type;
85 
86  gwinSetColor(ghc, White);
87  gwinPrintf(ghc, "This is detected as a %s device\n\n",
88  deviceType == GEVENT_MOUSE ? "MOUSE" : (pem->type == GEVENT_TOUCH ? "TOUCH" : "UNKNOWN"));
89 
90  if (ghNext)
91  gwinPrintf(ghc, "Press Next or Back to continue.\n");
92  else if (deviceType == GEVENT_MOUSE)
93  gwinPrintf(ghc, "Click the mouse button to move on to the next test.\n");
94  else
95  gwinPrintf(ghc, "Press and release your finger to move on to the next test.\n");
96 
97  while(1) {
98  pe = geventEventWait(&gl, TIME_INFINITE);
99  if (pe->type == GEVENT_GWIN_BUTTON) {
100  peb = (GEventGWinButton *)pe;
101  if (peb->button == ghPrev)
102  goto StepClickJitter;
103  if (peb->button == ghNext)
104  break;
105  }
106  if (pe->type == GEVENT_MOUSE || pe->type == GEVENT_TOUCH) {
107  pem = (GEventMouse *)pe;
108  if (!ghNext && (pem->meta & GMETA_MOUSE_UP))
109  break;
110  }
111  }
112 
113  /*
114  * Test: Mouse raw reading jitter
115  */
116 
117 StepRawJitter:
118  gwinClear(ghc);
119  gwinSetColor(ghc, Yellow);
120  gwinPrintf(ghc, "\n2. GINPUT_MOUSE_READ_CYCLES\n\n");
121 
122  gwinSetColor(ghc, White);
123  if (deviceType == GEVENT_MOUSE)
124  gwinPrintf(ghc, "Press and hold the mouse button.\n\n");
125  else
126  gwinPrintf(ghc, "Press and hold on the surface.\n\n");
127  gwinPrintf(ghc, "Numbers will display in this window.\n"
128  "Ensure that values don't jump around very much when your finger is stationary.\n\n"
129  "Increasing GINPUT_MOUSE_READ_CYCLES helps reduce jitter but increases CPU usage.\n\n");
130 
131  if (ghNext)
132  gwinPrintf(ghc, "Press Next or Back to continue.\n");
133  else if (deviceType == GEVENT_MOUSE)
134  gwinPrintf(ghc, "Release the mouse button to move on to the next test.\n");
135  else
136  gwinPrintf(ghc, "Release your finger to move on to the next test.\n");
137 
138  // For this test turn on ALL mouse movement events
139  geventAttachSource(&gl, gs, GLISTEN_MOUSEDOWNMOVES|GLISTEN_MOUSEMETA|GLISTEN_MOUSENOFILTER);
140 
141  while(1) {
142  pe = geventEventWait(&gl, TIME_INFINITE);
143  if (pe->type == GEVENT_GWIN_BUTTON) {
144  peb = (GEventGWinButton *)pe;
145  if (peb->button == ghPrev)
146  goto StepDeviceType;
147  if (peb->button == ghNext)
148  break;
149  }
150  if (pe->type == GEVENT_MOUSE || pe->type == GEVENT_TOUCH) {
151  pem = (GEventMouse *)pe;
152  if ((pem->current_buttons & GINPUT_MOUSE_BTN_LEFT))
153  gwinPrintf(ghc, "%u:%u\n", pem->x, pem->y);
154  if (!ghNext && (pem->meta & GMETA_MOUSE_UP))
155  break;
156  }
157  }
158 
159  // Reset to just changed movements.
160  geventAttachSource(&gl, gs, GLISTEN_MOUSEDOWNMOVES|GLISTEN_MOUSEMETA);
161 
162  /*
163  * Test: Calibration
164  */
165 
166 StepCalibrate:
167  gwinClear(ghc);
168  gwinSetColor(ghc, Yellow);
169  gwinPrintf(ghc, "\n3. GINPUT_MOUSE_CALIBRATION_ERROR\n\n");
170  gwinSetColor(ghc, Gray);
171  gwinPrintf(ghc, "Ensure GINPUT_MOUSE_NEED_CALIBRATION = TRUE and GINPUT_MOUSE_CALIBRATION_ERROR is >= 0\n\n");
172  gwinSetColor(ghc, White);
173  gwinPrintf(ghc, "You will be presented with a number of points to touch.\nPress them in turn.\n\n"
174  "If the calibration repeatedly fails, increase GINPUT_MOUSE_CALIBRATION_ERROR and try again.\n\n");
175 
176  if (ghNext)
177  gwinPrintf(ghc, "Press Next to start the calibration.\n");
178  else if (deviceType == GEVENT_MOUSE)
179  gwinPrintf(ghc, "Click the mouse button to start the calibration.\n");
180  else
181  gwinPrintf(ghc, "Press and release your finger to start the calibration.\n");
182 
183  while(1) {
184  pe = geventEventWait(&gl, TIME_INFINITE);
185  if (pe->type == GEVENT_GWIN_BUTTON) {
186  peb = (GEventGWinButton *)pe;
187  if (peb->button == ghPrev)
188  goto StepRawJitter;
189  if (peb->button == ghNext)
190  break;
191  }
192  if (pe->type == GEVENT_MOUSE || pe->type == GEVENT_TOUCH) {
193  pem = (GEventMouse *)pe;
194  if (!ghNext && (pem->meta & GMETA_MOUSE_UP))
195  break;
196  }
197  }
198 
199  // Calibrate
200  ginputCalibrateMouse(0);
201 
202  /* From now on we can use Next and Previous Buttons */
203  if (!ghNext) {
204  gwinAttachMouse(0);
205  gwinAttachListener(&gl);
206 
207  wi.customDraw = 0; wi.customParam = 0; wi.customStyle = 0;
208  wi.g.show = TRUE; wi.g.x = swidth-50; wi.g.y = 0; wi.g.width = 50; wi.g.height = 20;
209  wi.text = "Next"; ghNext = gwinButtonCreate(&gNext, &wi);
210 
211  wi.g.show = TRUE; wi.g.x = swidth-100;
212  wi.text = "Back"; ghPrev = gwinButtonCreate(&gPrev, &wi);
213  }
214 
215  // Calibration used the whole screen - re-establish our title
216  gdispFillStringBox(0, 0, swidth, 20, "Touch Calibration", font, Green, White, justifyLeft);
217  gwinRedraw(ghNext);
218  gwinRedraw(ghPrev);
219 
220  /*
221  * Test: Mouse movement jitter
222  */
223 
224 StepJitter:
225  gwinClear(ghc);
226  gwinSetColor(ghc, Yellow);
227  gwinPrintf(ghc, "\n4. GINPUT_MOUSE_MOVE_JITTER\n\n");
228 
229  gwinSetColor(ghc, White);
230  if (deviceType == GEVENT_MOUSE)
231  gwinPrintf(ghc, "Press and hold the mouse button and move around as if to draw.\n\n");
232  else
233  gwinPrintf(ghc, "Press firmly on the surface and move around as if to draw.\n\n");
234 
235  gwinPrintf(ghc, "Dots will display in this window. Ensure that when you stop moving your finger that "
236  "new dots stop displaying.\nNew dots should only display when your finger is moving.\n\n"
237  "Adjust GINPUT_MOUSE_MOVE_JITTER to the smallest value that this reliably works for.\n\n");
238  gwinPrintf(ghc, "Press Next or Back to continue.\n\n");
239 
240  while(1) {
241  pe = geventEventWait(&gl, TIME_INFINITE);
242  if (pe->type == GEVENT_GWIN_BUTTON) {
243  peb = (GEventGWinButton *)pe;
244  if (peb->button == ghPrev)
245  goto StepCalibrate;
246  if (peb->button == ghNext)
247  break;
248  }
249  if (pe->type == GEVENT_MOUSE || pe->type == GEVENT_TOUCH) {
250  pem = (GEventMouse *)pe;
251  if ((pem->current_buttons & GINPUT_MOUSE_BTN_LEFT))
252  gwinPrintf(ghc, ".");
253  }
254  }
255 
256  /*
257  * Test: Polling frequency
258  */
259 
260 StepPolling:
261  gwinClear(ghc);
262  gwinSetColor(ghc, Yellow);
263  gwinPrintf(ghc, "\n5. GINPUT_MOUSE_POLL_PERIOD\n\n");
264 
265  gwinSetColor(ghc, White);
266  gwinPrintf(ghc, "Press firmly on the surface (or press and hold the mouse button) and move around as if to draw.\n\n");
267  gwinPrintf(ghc, "A green line will follow your finger.\n"
268  "Adjust GINPUT_MOUSE_POLL_PERIOD to the highest value that provides a line without "
269  "gaps that are too big.\nDecreasing the value increases CPU usage.\n"
270  "About 25 (millisecs) normally produces good results."
271  "This test can be ignored for interrupt driven drivers.\n\n");
272  gwinPrintf(ghc, "Press Next or Back to continue.\n\n");
273 
274  while(1) {
275  pe = geventEventWait(&gl, TIME_INFINITE);
276  if (pe->type == GEVENT_GWIN_BUTTON) {
277  peb = (GEventGWinButton *)pe;
278  if (peb->button == ghPrev)
279  goto StepJitter;
280  if (peb->button == ghNext)
281  break;
282  }
283  if (pe->type == GEVENT_MOUSE || pe->type == GEVENT_TOUCH) {
284  pem = (GEventMouse *)pe;
285  if ((pem->current_buttons & GINPUT_MOUSE_BTN_LEFT))
286  gdispDrawPixel(pem->x, pem->y, Green);
287  }
288  }
289 
290  /*
291  * Test: Click Jitter
292  */
293 
294 StepClickJitter:
295  gwinClear(ghc);
296  gwinSetColor(ghc, Yellow);
297  gwinPrintf(ghc, "\n6. GINPUT_MOUSE_MAX_CLICK_JITTER\n\n");
298 
299  gwinSetColor(ghc, White);
300  gwinPrintf(ghc, "Press and release the touch surface to \"click\".\nTry both short and long presses.\n");
301  gwinPrintf(ghc, "For a mouse click with the left and right buttons.\n\n");
302  gwinPrintf(ghc, "Dots will display in this window. A yellow dash is a left (or short) click. "
303  "A red x is a right (or long) click.\n\n"
304  "Adjust GINPUT_MOUSE_CLICK_JITTER to the smallest value that this reliably works for.\n"
305  "Adjust GINPUT_MOUSE_CLICK_TIME to adjust distinguishing short vs long presses.\n"
306  "TIME_INFINITE means there are no long presses (although a right mouse button will still work).\n\n"
307  "Note: moving your finger (mouse) during a click cancels it.\n\n");
308  gwinPrintf(ghc, "This is the last test but you can press Next or Back to continue.\n\n");
309 
310  while(1) {
311  pe = geventEventWait(&gl, TIME_INFINITE);
312  if (pe->type == GEVENT_GWIN_BUTTON) {
313  peb = (GEventGWinButton *)pe;
314  if (peb->button == ghPrev)
315  goto StepPolling;
316  if (peb->button == ghNext)
317  break;
318  }
319  if (pe->type == GEVENT_MOUSE || pe->type == GEVENT_TOUCH) {
320  pem = (GEventMouse *)pe;
321  if ((pem->meta & GMETA_MOUSE_CLICK)) {
322  gwinSetColor(ghc, Yellow);
323  gwinPrintf(ghc, "-");
324  }
325  if ((pem->meta & GMETA_MOUSE_CXTCLICK)) {
326  gwinSetColor(ghc, Red);
327  gwinPrintf(ghc, "x");
328  }
329  }
330  }
331 
332  // Can't let this really exit
333  goto StepDeviceType;
334 }
