55 void main (void)
56 {
57  uint16_t xx, yy, colour;
58 
59  gfxInit();
60 
61  uint16_t width = (uint16_t)gdispGetWidth();
62  uint16_t height = (uint16_t)gdispGetHeight();
63 
64  float i=height/5+height%2+1, floorstart=height/5-1, spherespin=0.0,
65  l=width/2, m=height/4, n=.01*width, o=0.0, rotspeed=0.1, h, f, g;
66 
67  while (TRUE)
68  {
69  reset_cursor ();
70  StartStream ();
71 
72  for (xx=yy=0;
73  h = (m-yy)/i, f=-.3*(g=(l-xx)/i)+.954*h, yy<height;
74  yy += (xx = ++xx%width)==0 )
75  {
76  if (g*g < 1-h*h) /* if inside the ball */
77  if (((int)(9-spherespin+(.954*g+.3*h)/sqrtf(1-f*f))+(int)(2+f*2))%2==0)
78  colour = Red;
79  else
80  colour = White;
81  else
82  {
83  if (xx<floorstart || xx>width-floorstart)
84  colour = Darkgrey; /* side wall */
85  else
86  colour = Lightgrey; /* back wall */
87 
88  if (yy > height-floorstart)
89  if (xx < height-yy || height-yy > width-xx) /* floor */
90  colour = Darkgrey;
91  else
92  colour = Midgrey;
93 
94  if (g*(g+.6)+.09+h*h < 1)
95  colour >>= 1; /* ball shadow; make it darker */
96  }
97 
98  WriteStream (colour); /* pixel to the LCD */
99  }
100 
101  StopStream();
102  spherespin += rotspeed;
103  m += o;
104  o = m > height-1.75*floorstart ? -.04*height : o+.002*height;
105  n = (l+=n)<i || l>width-i ? rotspeed=-rotspeed,-n : n;
106  }
107 }
