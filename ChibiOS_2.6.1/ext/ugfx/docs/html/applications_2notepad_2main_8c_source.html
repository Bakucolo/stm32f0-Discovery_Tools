45 void drawScreen(void) {
46  char *msg = "uGFX";
47  font_t font1, font2;
48 
49  font1 = gdispOpenFont("DejaVuSans24*");
50  font2 = gdispOpenFont("DejaVuSans12*");
51 
52  gdispClear(White);
53  gdispDrawString(gdispGetWidth()-gdispGetStringWidth(msg, font1)-3, 3, msg, font1, Black);
54 
55  /* colors */
56  gdispFillArea(0 * COLOR_SIZE + 3, 3, COLOR_SIZE, COLOR_SIZE, Black); /* Black */
57  gdispFillArea(1 * COLOR_SIZE + 3, 3, COLOR_SIZE, COLOR_SIZE, Red); /* Red */
58  gdispFillArea(2 * COLOR_SIZE + 3, 3, COLOR_SIZE, COLOR_SIZE, Yellow); /* Yellow */
59  gdispFillArea(3 * COLOR_SIZE + 3, 3, COLOR_SIZE, COLOR_SIZE, Green); /* Green */
60  gdispFillArea(4 * COLOR_SIZE + 3, 3, COLOR_SIZE, COLOR_SIZE, Blue); /* Blue */
61  gdispDrawBox (5 * COLOR_SIZE + 3, 3, COLOR_SIZE, COLOR_SIZE, Black); /* White */
62 
63  /* pens */
64  gdispFillStringBox(OFFSET * 2, DRAW_PEN(1), PEN_SIZE, PEN_SIZE, "1", font2, White, Black, justifyCenter);
65  gdispFillStringBox(OFFSET * 2, DRAW_PEN(2), PEN_SIZE, PEN_SIZE, "2", font2, White, Black, justifyCenter);
66  gdispFillStringBox(OFFSET * 2, DRAW_PEN(3), PEN_SIZE, PEN_SIZE, "3", font2, White, Black, justifyCenter);
67  gdispFillStringBox(OFFSET * 2, DRAW_PEN(4), PEN_SIZE, PEN_SIZE, "4", font2, White, Black, justifyCenter);
68  gdispFillStringBox(OFFSET * 2, DRAW_PEN(5), PEN_SIZE, PEN_SIZE, "5", font2, White, Black, justifyCenter);
69 
70  gdispCloseFont(font1);
71  gdispCloseFont(font2);
72 }

76 int main(void) {
77  color_t color = Black;
78  uint16_t pen = 0;
79 
80  gfxInit();
81  ginputGetMouse(0);
82  gdispSetOrientation(GDISP_ROTATE_90);
83 
84  drawScreen();
85 
86  while (TRUE) {
87  ginputGetMouseStatus(0, &ev);
88  if (!(ev.current_buttons & GINPUT_MOUSE_BTN_LEFT))
89  continue;
90 
91  /* inside color box ? */
92  if(ev.y >= OFFSET && ev.y <= COLOR_SIZE) {
93  if(GET_COLOR(0)) color = Black;
94  else if(GET_COLOR(1)) color = Red;
95  else if(GET_COLOR(2)) color = Yellow;
96  else if(GET_COLOR(3)) color = Green;
97  else if(GET_COLOR(4)) color = Blue;
98  else if(GET_COLOR(5)) color = White;
99 
100  /* inside pen box ? */
101  } else if(ev.x >= OFFSET && ev.x <= PEN_SIZE) {
102  if(GET_PEN(1)) pen = 0;
103  else if(GET_PEN(2)) pen = 1;
104  else if(GET_PEN(3)) pen = 2;
105  else if(GET_PEN(4)) pen = 3;
106  else if(GET_PEN(5)) pen = 4;
107 
108  /* inside drawing area ? */
109  } else if(DRAW_AREA(ev.x, ev.y)) {
110  if(pen == 0)
111  gdispDrawPixel(ev.x, ev.y, color);
112  else
113  gdispFillCircle(ev.x, ev.y, pen, color);
114  }
115  }
116 }
