30 /**
31  * This demo demonstrates the use of the GADC module using it read both a microphone,
32  * an analogue dial wheel and a temperature sensor.
33  * The microphone gets read at high frequency to display a very simple oscilloscope.
34  * The dial and temperature gets read at a low frequency to just print when
35  * it changes value.
36  *
37  * It also demonstrates how to write your own custom GWIN window type.
38  */
39 #include "gfx.h"
40 
41 /* Include our custom gwin oscilloscope */
42 #include "gwinosc.h"
43 
44 /*
45  * Match these to your hardware
46  * If you don't have a DIAL device or a TEMP device - just don't define it.
47  */
48 #define MY_MIC_DEVICE GADC_PHYSDEV_MICROPHONE
49 #define MY_DIAL_DEVICE GADC_PHYSDEV_DIAL
50 #define MY_TEMP_DEVICE GADC_PHYSDEV_TEMPERATURE
51 #define MY_DIAL_JITTER 1
52 #define MY_TEMP_JITTER 3
53 
54 /* Specify our timing parameters */
55 #define MY_MIC_FREQUENCY 4000 /* 4khz */
56 #define MY_LS_DELAY 200 /* 200ms (5 times per second) for the dial and temperature */
57 
58 /* The desired size for our scope window */
59 #define SCOPE_CX 64
60 #define SCOPE_CY 64
61 
62 /* Data */
63 static GScopeObject gScopeWindow;
64 static GConsoleObject gTextWindow;
65 static GTimer lsTimer;
66 
67 #ifdef MY_DIAL_DEVICE
68  static adcsample_t dialvalue;
69  static adcsample_t lastdial = -(MY_DIAL_JITTER+1);
70 
71  /**
72  * We have got a dial reading - handle it
73  */
74  static void GotDialReading(adcsample_t *buffer, void *param) {
75  (void) buffer;
76 
77  /* Buffer should always point to "dialvalue" anyway */
78 
79  /* Remove jitter from the value */
80  if ((dialvalue > lastdial && dialvalue - lastdial > MY_DIAL_JITTER)
81  || (lastdial > dialvalue && lastdial - dialvalue > MY_DIAL_JITTER)) {
82 
83  /* Write the value */
84  gwinPrintf((GHandle)param, "DIAL: %u\n", dialvalue);
85 
86  /* Save for next time */
87  lastdial = dialvalue;
88  }
89  }
90 #endif
91 
92 #ifdef MY_TEMP_DEVICE
93  static adcsample_t tempvalue;
94  static adcsample_t lasttemp = -(MY_TEMP_JITTER+1);
95 
96  /**
97  * We have got a temperature reading - handle it
98  */
99  static void GotTempReading(adcsample_t *buffer, void *param) {
100  (void) buffer;
101 
102  /* Buffer should always point to "tempvalue" anyway */
103 
104  /* Remove jitter from the value */
105  if ((tempvalue > lasttemp && tempvalue - lasttemp > MY_TEMP_JITTER)
106  || (lasttemp > tempvalue && lasttemp - tempvalue > MY_TEMP_JITTER)) {
107 
108  /* Write the value */
109  gwinPrintf((GHandle)param, "TEMP: %u\n", tempvalue);
110 
111  /* Save for next time */
112  lasttemp = tempvalue;
113  }
114  }
115 #endif
116 
117 #if defined(MY_DIAL_DEVICE) || defined(MY_TEMP_DEVICE)
118  /**
119  * Start a read of the dial and temperature
120  */
121  static void LowSpeedTimer(void *param) {
122  /* We are not checking for an error here - but who cares, this is just a demo */
123  #ifdef MY_DIAL_DEVICE
124  gadcLowSpeedStart(MY_DIAL_DEVICE, &dialvalue, GotDialReading, param);
125  #endif
126  #ifdef MY_TEMP_DEVICE
127  gadcLowSpeedStart(MY_TEMP_DEVICE, &tempvalue, GotTempReading, param);
128  #endif
129  }
130 #endif

132 /*
133  * Application entry point.
134  */
135 int main(void) {
136  GHandle ghScope;
137  coord_t swidth, sheight;
138  #if defined(MY_DIAL_DEVICE) || defined(MY_TEMP_DEVICE)
139  GHandle ghText;
140  font_t font;
141  #endif
142 
143  gfxInit();
144 
145  /* Get the screen dimensions */
146  swidth = gdispGetWidth();
147  sheight = gdispGetHeight();
148 
149  #if defined(MY_DIAL_DEVICE) || defined(MY_TEMP_DEVICE)
150  /* Set up the console window we use for dial readings */
151  font = gdispOpenFont("UI2");
152  gwinSetDefaultFont(font);
153  {
154  GWindowInit wi;
155  wi.show = TRUE;
156  wi.x = wi.y = 0;
157  wi.width = swidth-SCOPE_CX;
158  wi.height = sheight;
159  ghText = gwinConsoleCreate(&gTextWindow, &wi);
160  }
161  gwinSetBgColor(ghText, Black);
162  gwinSetColor(ghText, Yellow);
163  gwinClear(ghText);
164 
165  /* Start our timer for reading the dial */
166  gtimerInit(&lsTimer);
167  gtimerStart(&lsTimer, LowSpeedTimer, ghText, TRUE, MY_LS_DELAY);
168  #endif
169 
170  /* Set up the scope window in the top right on the screen */
171  {
172  GWindowInit wi;
173  wi.show = TRUE;
174  wi.x = swidth-SCOPE_CX;
175  wi.y = 0;
176  wi.width = SCOPE_CX;
177  wi.height = SCOPE_CY;
178  ghScope = gwinScopeCreate(&gScopeWindow, &wi, MY_MIC_DEVICE, MY_MIC_FREQUENCY);
179  }
180  gwinSetBgColor(ghScope, White);
181  gwinSetColor(ghScope, Red);
182  gwinClear(ghScope);
183 
184  /* Just keep displaying the scope traces */
185  while (TRUE) {
186  /**
187  * The function below internally performs a wait thus giving the timer thread a
188  * chance to run.
189  */
190  gwinScopeWaitForTrace(ghScope);
191  }
192 }
