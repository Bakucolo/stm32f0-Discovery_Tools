206 /**
207  * Set the visibility of widgets based on which tab is selected.
208  */
209 static void setTab(GHandle tab) {
210  /* Make sure everything is invisible first */
211  gwinSetVisible(ghButton1, FALSE); gwinSetVisible(ghButton2, FALSE);
212  gwinSetVisible(ghButton3, FALSE); gwinSetVisible(ghButton4, FALSE);
213  gwinSetVisible(ghSlider1, FALSE); gwinSetVisible(ghSlider2, FALSE);
214  gwinSetVisible(ghSlider3, FALSE); gwinSetVisible(ghSlider4, FALSE);
215  gwinSetVisible(ghCheckbox1, FALSE); gwinSetVisible(ghCheckbox2, FALSE); gwinSetVisible(ghCheckDisableAll, FALSE);
216  gwinSetVisible(ghLabel1, FALSE);
217  gwinSetVisible(ghRadio1, FALSE); gwinSetVisible(ghRadio2, FALSE);
218  gwinSetVisible(ghRadioWhite, FALSE);gwinSetVisible(ghRadioBlack, FALSE);gwinSetVisible(ghRadioYellow, FALSE);
219  gwinSetVisible(ghList1, FALSE); gwinSetVisible(ghList2, FALSE); gwinSetVisible(ghList3, FALSE);
220  gwinSetVisible(ghImage1, FALSE);
221 
222  /* Turn on widgets depending on the tab selected */
223  if (tab == ghTabButtons) {
224  gwinSetVisible(ghButton1, TRUE); gwinSetVisible(ghButton2, TRUE);
225  gwinSetVisible(ghButton3, TRUE); gwinSetVisible(ghButton4, TRUE);
226  } else if (tab == ghTabSliders) {
227  gwinSetVisible(ghSlider1, TRUE); gwinSetVisible(ghSlider2, TRUE);
228  gwinSetVisible(ghSlider3, TRUE); gwinSetVisible(ghSlider4, TRUE);
229  } else if (tab == ghTabCheckboxes) {
230  gwinSetVisible(ghCheckbox1, TRUE); gwinSetVisible(ghCheckbox2, TRUE); gwinSetVisible(ghCheckDisableAll, TRUE);
231  } else if (tab == ghTabLabels) {
232  gwinSetVisible(ghLabel1, TRUE);
233  } else if (tab == ghTabRadios) {
234  gwinSetVisible(ghRadio1, TRUE); gwinSetVisible(ghRadio2, TRUE);
235  gwinSetVisible(ghRadioWhite, TRUE); gwinSetVisible(ghRadioBlack, TRUE); gwinSetVisible(ghRadioYellow, TRUE);
236  } else if (tab == ghTabLists) {
237  gwinSetVisible(ghList1, TRUE); gwinSetVisible(ghList2, TRUE); gwinSetVisible(ghList3, TRUE);
238  } else if (tab == ghTabImages) {
239  gwinSetVisible(ghImage1, TRUE);
240  }
241 }

259 int main(void) {
260  GEvent * pe;
261 
262  // Initialize the display
263  gfxInit();
264 
265  // Set the widget defaults
266  gwinSetDefaultFont(gdispOpenFont("*"));
267  gwinSetDefaultStyle(&WhiteWidgetStyle, FALSE);
268  gdispClear(White);
269 
270  // Connect the mouse
271  #if GINPUT_NEED_MOUSE
272  gwinAttachMouse(0);
273  #endif
274 
275  #if GDISP_NEED_CONTROL
276  gdispSetOrientation(GDISP_ROTATE_90);
277  #endif
278 
279  // Create the gwin windows/widgets
280  createWidgets();
281 
282  // Assign toggles and dials to specific buttons & sliders etc.
283  #if GINPUT_NEED_TOGGLE
284  gwinAttachToggle(ghButton1, 0, 0);
285  gwinAttachToggle(ghButton2, 0, 1);
286  #endif
287  #if GINPUT_NEED_DIAL
288  gwinAttachDial(ghSlider1, 0, 0);
289  gwinAttachDial(ghSlider3, 0, 1);
290  #endif
291 
292  // Make the console visible
293  gwinSetVisible(ghConsole, TRUE);
294  gwinClear(ghConsole);
295 
296  // We want to listen for widget events
297  geventListenerInit(&gl);
298  gwinAttachListener(&gl);
299 
300  // Press the Tab we want visible
301  gwinRadioPress(ghTabButtons);
302 
303  while(1) {
304  // Get an Event
305  pe = geventEventWait(&gl, TIME_INFINITE);
306 
307  switch(pe->type) {
308  case GEVENT_GWIN_BUTTON:
309  gwinPrintf(ghConsole, "Button %s\n", gwinGetText(((GEventGWinButton *)pe)->button));
310  break;
311 
312  case GEVENT_GWIN_SLIDER:
313  gwinPrintf(ghConsole, "Slider %s=%d\n", gwinGetText(((GEventGWinSlider *)pe)->slider), ((GEventGWinSlider *)pe)->position);
314  break;
315 
316  case GEVENT_GWIN_CHECKBOX:
317  gwinPrintf(ghConsole, "Checkbox %s=%s\n", gwinGetText(((GEventGWinCheckbox *)pe)->checkbox), ((GEventGWinCheckbox *)pe)->isChecked ? "Checked" : "UnChecked");
318 
319  // If it is the Disable All checkbox then do that.
320  if (((GEventGWinCheckbox *)pe)->checkbox == ghCheckDisableAll) {
321  gwinPrintf(ghConsole, "%s All\n", ((GEventGWinCheckbox *)pe)->isChecked ? "Disable" : "Enable");
322  setEnabled(!((GEventGWinCheckbox *)pe)->isChecked);
323  }
324  break;
325 
326  case GEVENT_GWIN_LIST:
327  gwinPrintf(ghConsole, "List %s Item %d %s\n", gwinGetText(((GEventGWinList *)pe)->list), ((GEventGWinList *)pe)->item,
328  gwinListItemIsSelected(((GEventGWinList *)pe)->list, ((GEventGWinList *)pe)->item) ? "Selected" : "Unselected");
329  break;
330 
331  case GEVENT_GWIN_RADIO:
332  gwinPrintf(ghConsole, "Radio Group %u=%s\n", ((GEventGWinRadio *)pe)->group, gwinGetText(((GEventGWinRadio *)pe)->radio));
333 
334  switch(((GEventGWinRadio *)pe)->group) {
335  case GROUP_TABS:
336 
337  // Set control visibility depending on the tab selected
338  setTab(((GEventGWinRadio *)pe)->radio);
339 
340  // Do some special animation for Label1 to demonstrate auto width sizing
341  if (((GEventGWinRadio *)pe)->radio == ghTabLabels) {
342  gwinPrintf(ghConsole, "Change Label Text\n");
343  gfxSleepMilliseconds(1000);
344  gwinSetText(ghLabel1, "Very Big Label", FALSE);
345 
346  gfxSleepMilliseconds(1000);
347  gwinSetText(ghLabel1, "Label", FALSE);
348  }
349  break;
350 
351  case GROUP_COLORS:
352  {
353  const GWidgetStyle *pstyle;
354 
355  gwinPrintf(ghConsole, "Change Color Scheme\n");
356 
357  if (((GEventGWinRadio *)pe)->radio == ghRadioYellow)
358  pstyle = &YellowWidgetStyle;
359  else if (((GEventGWinRadio *)pe)->radio == ghRadioBlack)
360  pstyle = &BlackWidgetStyle;
361  else
362  pstyle = &WhiteWidgetStyle;
363 
364  // Clear the screen to the new color - we avoid the console area as it can't redraw itself
365  #if GDISP_NEED_CLIP
366  gdispUnsetClip();
367  #endif
368  gdispFillArea(0, 0, ScrWidth, ScrHeight/2, pstyle->background);
369  gdispFillArea(0, ScrHeight/2, ScrWidth/2, ScrHeight/2, pstyle->background);
370 
371  // Update the style on all controls
372  gwinSetDefaultStyle(pstyle, TRUE);
373  }
374  break;
375  }
376  break;
377 
378  default:
379  gwinPrintf(ghConsole, "Unknown %d\n", pe->type);
380  break;
381  }
382  }
383  return 0;
384 }
