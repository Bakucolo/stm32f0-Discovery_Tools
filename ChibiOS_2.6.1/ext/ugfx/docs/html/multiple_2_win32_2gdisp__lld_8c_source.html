82 static LRESULT myWindowProc(HWND hWnd, UINT Msg, WPARAM wParam, LPARAM lParam)
83 {
84  HDC dc;
85  PAINTSTRUCT ps;
86  #if GINPUT_NEED_TOGGLE
87  HBRUSH hbrOn, hbrOff;
88  HPEN pen;
89  RECT rect;
90  HGDIOBJ old;
91  POINT p;
92  coord_t pos;
93  uint8_t bit;
94  #endif
95 
96  switch (Msg) {
97  case WM_CREATE:
98  break;
99  case WM_LBUTTONDOWN:
100  #if GINPUT_NEED_MOUSE
101  if ((coord_t)HIWORD(lParam) < wHeight) {
102  mousebuttons |= GINPUT_MOUSE_BTN_LEFT;
103  goto mousemove;
104  }
105  #endif
106  #if GINPUT_NEED_TOGGLE
107  bit = 1 << ((coord_t)LOWORD(lParam)*8/wWidth);
108  toggles ^= bit;
109  rect.left = 0;
110  rect.right = wWidth;
111  rect.top = wHeight;
112  rect.bottom = wHeight + WIN32_BUTTON_AREA;
113  InvalidateRect(hWnd, &rect, FALSE);
114  UpdateWindow(hWnd);
115  #if GINPUT_TOGGLE_POLL_PERIOD == TIME_INFINITE
116  ginputToggleWakeup();
117  #endif
118  #endif
119  break;
120  case WM_LBUTTONUP:
121  #if GINPUT_NEED_TOGGLE
122  if ((toggles & 0x0F)) {
123  toggles &= ~0x0F;
124  rect.left = 0;
125  rect.right = wWidth;
126  rect.top = wHeight;
127  rect.bottom = wHeight + WIN32_BUTTON_AREA;
128  InvalidateRect(hWnd, &rect, FALSE);
129  UpdateWindow(hWnd);
130  #if GINPUT_TOGGLE_POLL_PERIOD == TIME_INFINITE
131  ginputToggleWakeup();
132  #endif
133  }
134  #endif
135  #if GINPUT_NEED_MOUSE
136  if ((coord_t)HIWORD(lParam) < wHeight) {
137  mousebuttons &= ~GINPUT_MOUSE_BTN_LEFT;
138  goto mousemove;
139  }
140  #endif
141  break;
142 #if GINPUT_NEED_MOUSE
143  case WM_MBUTTONDOWN:
144  if ((coord_t)HIWORD(lParam) < wHeight) {
145  mousebuttons |= GINPUT_MOUSE_BTN_MIDDLE;
146  goto mousemove;
147  }
148  break;
149  case WM_MBUTTONUP:
150  if ((coord_t)HIWORD(lParam) < wHeight) {
151  mousebuttons &= ~GINPUT_MOUSE_BTN_MIDDLE;
152  goto mousemove;
153  }
154  break;
155  case WM_RBUTTONDOWN:
156  if ((coord_t)HIWORD(lParam) < wHeight) {
157  mousebuttons |= GINPUT_MOUSE_BTN_RIGHT;
158  goto mousemove;
159  }
160  break;
161  case WM_RBUTTONUP:
162  if ((coord_t)HIWORD(lParam) < wHeight) {
163  mousebuttons &= ~GINPUT_MOUSE_BTN_RIGHT;
164  goto mousemove;
165  }
166  break;
167  case WM_MOUSEMOVE:
168  if ((coord_t)HIWORD(lParam) >= wHeight)
169  break;
170  mousemove:
171  mousex = (coord_t)LOWORD(lParam);
172  mousey = (coord_t)HIWORD(lParam);
173  #if GINPUT_MOUSE_POLL_PERIOD == TIME_INFINITE
174  ginputMouseWakeup();
175  #endif
176  break;
177 #endif
178  case WM_SYSKEYDOWN:
179  case WM_KEYDOWN:
180  case WM_SYSKEYUP:
181  case WM_KEYUP:
182  break;
183  case WM_CHAR:
184  case WM_DEADCHAR:
185  case WM_SYSCHAR:
186  case WM_SYSDEADCHAR:
187  break;
188  case WM_PAINT:
189  dc = BeginPaint(hWnd, &ps);
190  BitBlt(dc, ps.rcPaint.left, ps.rcPaint.top,
191  ps.rcPaint.right - ps.rcPaint.left,
192  (ps.rcPaint.bottom > wHeight ? wHeight : ps.rcPaint.bottom) - ps.rcPaint.top,
193  dcBuffer, ps.rcPaint.left, ps.rcPaint.top, SRCCOPY);
194  #if GINPUT_NEED_TOGGLE
195  if (ps.rcPaint.bottom >= wHeight) {
196  pen = CreatePen(PS_SOLID, 1, COLOR2BGR(Black));
197  hbrOn = CreateSolidBrush(COLOR2BGR(Blue));
198  hbrOff = CreateSolidBrush(COLOR2BGR(Gray));
199  old = SelectObject(dc, pen);
200  MoveToEx(dc, 0, wHeight, &p);
201  LineTo(dc, wWidth, wHeight);
202  for(pos = 0, bit=1; pos < wWidth; pos=rect.right, bit <<= 1) {
203  rect.left = pos;
204  rect.right = pos + wWidth/8;
205  rect.top = wHeight;
206  rect.bottom = wHeight + WIN32_BUTTON_AREA;
207  FillRect(dc, &rect, (toggles & bit) ? hbrOn : hbrOff);
208  if (pos > 0) {
209  MoveToEx(dc, rect.left, rect.top, &p);
210  LineTo(dc, rect.left, rect.bottom);
211  }
212  }
213  DeleteObject(hbrOn);
214  DeleteObject(hbrOff);
215  SelectObject(dc, old);
216  }
217  #endif
218  EndPaint(hWnd, &ps);
219  break;
220  case WM_DESTROY:
221  PostQuitMessage(0);
222  SelectObject(dcBuffer, dcOldBitmap);
223  DeleteDC(dcBuffer);
224  DeleteObject(dcBitmap);
225  winRootWindow = NULL;
226  break;
227  default:
228  return DefWindowProc(hWnd, Msg, wParam, lParam);
229  }
230  return 0;
231 }

260

263  wHeight = rect.bottom - rect.top - WIN32_BUTTON_AREA;

271  ShowWindow(winRootWindow, SW_SHOW);
272  UpdateWindow(winRootWindow);

303 /**
304  * @brief Low level GDISP driver initialisation.
305  * @return TRUE if successful, FALSE on error.
306  *
307  * @notapi
308  */
309 bool_t gdisp_lld_init(void) {
310  RECT rect;
311  gfxThreadHandle hth;
312 
313  /* Set the window dimensions */
314  GetWindowRect(GetDesktopWindow(), &rect);
315  wWidth = rect.right - rect.left;
316  wHeight = rect.bottom - rect.top - WIN32_BUTTON_AREA;
317  if (wWidth > GDISP_SCREEN_WIDTH)
318  wWidth = GDISP_SCREEN_WIDTH;
319  if (wHeight > GDISP_SCREEN_HEIGHT)
320  wHeight = GDISP_SCREEN_HEIGHT;
321 
322  /* Initialise the window */
323  if (!(hth = gfxThreadCreate(waWindowThread, sizeof(waWindowThread), HIGH_PRIORITY, WindowThread, 0))) {
324  fprintf(stderr, "Cannot create window thread\n");
325  exit(-1);
326  }
327  gfxThreadClose(hth);
328  while (!isReady)
329  Sleep(1);
330 
331  /* Initialise the GDISP structure to match */
332  GDISP.Orientation = GDISP_ROTATE_0;
333  GDISP.Powermode = powerOn;
334  GDISP.Backlight = 100;
335  GDISP.Contrast = 50;
336  GDISP.Width = wWidth;
337  GDISP.Height = wHeight;
338  #if GDISP_NEED_VALIDATION || GDISP_NEED_CLIP
339  GDISP.clipx0 = 0;
340  GDISP.clipy0 = 0;
341  GDISP.clipx1 = GDISP.Width;
342  GDISP.clipy1 = GDISP.Height;
343  #endif
344  return TRUE;
345 }

347 /**
348  * @brief Draws a pixel on the display.
349  *
350  * @param[in] x X location of the pixel
351  * @param[in] y Y location of the pixel
352  * @param[in] color The color of the pixel
353  *
354  * @notapi
355  */
356 void gdisp_lld_draw_pixel(coord_t x, coord_t y, color_t color) {
357  HDC dc;
358  #if WIN32_USE_MSG_REDRAW
359  RECT rect;
360  #endif
361  #if GDISP_NEED_CONTROL
362  coord_t t;
363  #endif
364 
365  #if GDISP_NEED_VALIDATION || GDISP_NEED_CLIP
366  // Clip pre orientation change
367  if (x < GDISP.clipx0 || y < GDISP.clipy0 || x >= GDISP.clipx1 || y >= GDISP.clipy1) return;
368  #endif
369 
370  #if GDISP_NEED_CONTROL
371  switch(GDISP.Orientation) {
372  case GDISP_ROTATE_0:
373  break;
374  case GDISP_ROTATE_90:
375  t = GDISP.Height - 1 - y;
376  y = x;
377  x = t;
378  break;
379  case GDISP_ROTATE_180:
380  x = GDISP.Width - 1 - x;
381  y = GDISP.Height - 1 - y;
382  break;
383  case GDISP_ROTATE_270:
384  t = GDISP.Width - 1 - x;
385  x = y;
386  y = t;
387  break;
388  }
389  #endif
390 
391  // Draw the pixel in the buffer
392  color = COLOR2BGR(color);
393  SetPixel(dcBuffer, x, y, color);
394 
395  #if WIN32_USE_MSG_REDRAW
396  rect.left = x; rect.right = x+1;
397  rect.top = y; rect.bottom = y+1;
398  InvalidateRect(winRootWindow, &rect, FALSE);
399  UpdateWindow(winRootWindow);
400  #else
401  // Draw the pixel again directly on the screen.
402  // This is cheaper than invalidating a single pixel in the window
403  dc = GetDC(winRootWindow);
404  SetPixel(dc, x, y, color);
405  ReleaseDC(winRootWindow, dc);
406  #endif
407 }

552  RECT rect;
553  HBRUSH hbr;
554 
555  #if GDISP_NEED_VALIDATION || GDISP_NEED_CLIP

562  #endif
563 
564  #if GDISP_NEED_CONTROL
565  switch(GDISP.Orientation) {
566  case GDISP_ROTATE_0:
567  rect.top = y;
568  rect.bottom = rect.top+cy;
569  rect.left = x;
570  rect.right = rect.left+cx;
571  break;
572  case GDISP_ROTATE_90:
573  rect.top = x;
574  rect.bottom = rect.top+cx;
575  rect.right = GDISP.Height - y;
576  rect.left = rect.right-cy;
577  break;

580  rect.top = rect.bottom-cy;
581  rect.right = GDISP.Width - x;
582  rect.left = rect.right-cx;
583  break;
584  case GDISP_ROTATE_270:
585  rect.bottom = GDISP.Width - x;
586  rect.top = rect.bottom-cx;
587  rect.left = y;
588  rect.right = rect.left+cy;
589  break;
590  }
591  #else
592  rect.top = y;
593  rect.bottom = rect.top+cy;
594  rect.left = x;

608  #else
