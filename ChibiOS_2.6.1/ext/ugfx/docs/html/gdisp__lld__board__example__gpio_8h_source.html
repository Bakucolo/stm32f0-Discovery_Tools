81  IOBus busDATA = {GDISP_CMD_PORT, 0xFFFFF, 0};

141 __inline void write_stream(uint16_t *buffer, uint16_t size) {
142  uint16_t i;
143 
144  Set_CS; Clr_RS; Set_WR; Clr_RD;
145 
146  for (i = 0; i < size; i++) {
147  Set_WR;
148  palWritePort(GDISP_DATA_PORT, buffer[i]);
149  Clr_WR;
150  }
151 
152  Clr_CS;
153 }

155 __inline void read_stream(uint16_t *buffer, size_t size) {
156  uint16_t i;
157 
158  Set_CS; Clr_RS; Clr_WR; Set_RD;
159 
160  for (i = 0; i < size; i++) {
161  Set_RD;
162  buffer[i] = palReadPort(GDISP_DATA_PORT);
163  Clr_RD;
164  }
