45 #if GDISP_NEED_ASYNC
46  static DECLARE_THREAD_FUNCTION(GDISPThreadHandler, arg) {
47  (void)arg;
48  gdisp_lld_msg_t *pmsg;
49 
50  while(1) {
51  /* Wait for msg with work to do. */
52  pmsg = (gdisp_lld_msg_t *)gfxQueueGet(&gdispQueue, TIME_INFINITE);
53 
54  /* OK - we need to obtain the mutex in case a synchronous operation is occurring */
55  gfxMutexEnter(&gdispMutex);
56 
57  gdisp_lld_msg_dispatch(pmsg);
58 
59  /* Mark the message as free */
60  pmsg->action = GDISP_LLD_MSG_NOP;
61 
62  gfxMutexExit(&gdispMutex);
63  }
64  return 0;
65  }
66 
67  static gdisp_lld_msg_t *gdispAllocMsg(gdisp_msgaction_t action) {
68  gdisp_lld_msg_t *p;
69 
70  while(1) { /* To be sure, to be sure */
71 
72  /* Wait for a slot */
73  gfxSemWait(&gdispMsgsSem, TIME_INFINITE);
74 
75  /* Find the slot */
76  gfxMutexEnter(&gdispMsgsMutex);
77  for(p=gdispMsgs; p < &gdispMsgs[GDISP_QUEUE_SIZE]; p++) {
78  if (p->action == GDISP_LLD_MSG_NOP) {
79  /* Allocate it */
80  p->action = action;
81  gfxMutexExit(&gdispMsgsMutex);
82  return p;
83  }
84  }
85  gfxMutexExit(&gdispMsgsMutex);
86 
87  /* Oops - none found, try again */
88  gfxSemSignal(&gdispMsgsSem);
89  }
90  }

108 #elif GDISP_NEED_ASYNC
109  void _gdispInit(void) {
110  unsigned i;
111  gfxThreadHandle hth;
112 
113  /* Mark all the Messages as free */
114  for(i=0; i < GDISP_QUEUE_SIZE; i++)
115  gdispMsgs[i].action = GDISP_LLD_MSG_NOP;
116 
117  /* Initialise our Queue, Mutex's and Counting Semaphore.
118  * A Mutex is required as well as the Queue and Thread because some calls have to be synchronous.
119  * Synchronous calls get handled by the calling thread, asynchronous by our worker thread.
120  */
121  gfxQueueInit(&gdispQueue);
122  gfxMutexInit(&gdispMutex);
123  gfxMutexInit(&gdispMsgsMutex);
124  gfxSemInit(&gdispMsgsSem, GDISP_QUEUE_SIZE, GDISP_QUEUE_SIZE);
125 
126  hth = gfxThreadCreate(waGDISPThread, sizeof(waGDISPThread), GDISP_THREAD_PRIORITY, GDISPThreadHandler, NULL);
127  if (hth) gfxThreadClose(hth);
128 
129  /* Initialise driver - synchronous */
130  gfxMutexEnter(&gdispMutex);
131  gdisp_lld_init();
132  gfxMutexExit(&gdispMutex);
133  }

136  gdisp_lld_init();
137  }

460 void gdispDrawBox(coord_t x, coord_t y, coord_t cx, coord_t cy, color_t color) {
461  /* No mutex required as we only call high level functions which have their own mutex */
462  coord_t x1, y1;
463 
464  x1 = x+cx-1;
465  y1 = y+cy-1;
466 
467  if (cx > 2) {
468  if (cy >= 1) {
469  gdispDrawLine(x, y, x1, y, color);
470  if (cy >= 2) {
471  gdispDrawLine(x, y1, x1, y1, color);
472  if (cy > 2) {
473  gdispDrawLine(x, y+1, x, y1-1, color);
474  gdispDrawLine(x1, y+1, x1, y1-1, color);
475  }
476  }
477  }
478  } else if (cx == 2) {
479  gdispDrawLine(x, y, x, y1, color);
480  gdispDrawLine(x1, y, x1, y1, color);
481  } else if (cx == 1) {
482  gdispDrawLine(x, y, x, y1, color);
483  }
484 }

496  void gdispFillConvexPoly(coord_t tx, coord_t ty, const point *pntarray, unsigned cnt, color_t color) {
497  const point *lpnt, *rpnt, *epnts;
498  fixed lx, rx, lk, rk;
499  coord_t y, ymax, lxc, rxc;
500 
501  epnts = &pntarray[cnt-1];
502 
503  /* Find a top point */
504  rpnt = pntarray;
505  for(lpnt=pntarray+1; lpnt <= epnts; lpnt++) {
506  if (lpnt->y < rpnt->y)
507  rpnt = lpnt;
508  }
509  lx = rx = FIXED(rpnt->x);
510  y = rpnt->y;
511 
512  /* Work out the slopes of the two attached line segs */
513  lpnt = rpnt <= pntarray ? epnts : rpnt-1;
514  while (lpnt->y == y) {
515  lx = FIXED(lpnt->x);
516  lpnt = lpnt <= pntarray ? epnts : lpnt-1;
517  if (!cnt--) return;
518  }
519  rpnt = rpnt >= epnts ? pntarray : rpnt+1;
520  while (rpnt->y == y) {
521  rx = rpnt->x<<16;
522  rpnt = rpnt >= epnts ? pntarray : rpnt+1;
523  if (!cnt--) return;
524  }
525  lk = (FIXED(lpnt->x) - lx) / (lpnt->y - y);
526  rk = (FIXED(rpnt->x) - rx) / (rpnt->y - y);
527 
528  while(1) {
529  /* Determine our boundary */
530  ymax = rpnt->y < lpnt->y ? rpnt->y : lpnt->y;
531 
532  /* Scan down the line segments until we hit a boundary */
533  for(; y < ymax; y++) {
534  lxc = NONFIXED(lx);
535  rxc = NONFIXED(rx);
536  /*
537  * Doesn't print the right hand point in order to allow polygon joining.
538  * Also ensures that we draw from left to right with the minimum number
539  * of pixels.
540  */
541  if (lxc < rxc) {
542  if (rxc - lxc == 1)
543  gdispDrawPixel(tx+lxc, ty+y, color);
544  else
545  gdispDrawLine(tx+lxc, ty+y, tx+rxc-1, ty+y, color);
546  } else if (lxc > rxc) {
547  if (lxc - rxc == 1)
548  gdispDrawPixel(tx+rxc, ty+y, color);
549  else
550  gdispDrawLine(tx+rxc, ty+y, tx+lxc-1, ty+y, color);
551  }
552 
553  lx += lk;
554  rx += rk;
555  }
556 
557  if (!cnt--) return;
558 
559  /* Replace the appropriate point */
560  if (ymax == lpnt->y) {
561  lpnt = lpnt <= pntarray ? epnts : lpnt-1;
562  while (lpnt->y == y) {
563  lx = FIXED(lpnt->x);
564  lpnt = lpnt <= pntarray ? epnts : lpnt-1;
565  if (!cnt--) return;
566  }
567  lk = (FIXED(lpnt->x) - lx) / (lpnt->y - y);
568  } else {
569  rpnt = rpnt >= epnts ? pntarray : rpnt+1;
570  while (rpnt->y == y) {
571  rx = FIXED(rpnt->x);
572  rpnt = rpnt >= epnts ? pntarray : rpnt+1;
573  if (!cnt--) return;
574  }
575  rk = (FIXED(rpnt->x) - rx) / (rpnt->y - y);
576  }
577  }
578  }
