50 /**
51  * @brief Write a value into a certain register
52  *
53  * @param[in] reg The register address
54  * @param[in] n The amount of bytes (one or two)
55  * @param[in] val The value
56  *
57  * @notapi
58  */
59 static void write_reg(uint8_t reg, uint8_t n, uint16_t val)
60 {
61  uint8_t txbuf[3];
62 
63  i2cAcquireBus(&I2CD1);
64 
65  txbuf[0] = reg;
66 
67  if (n == 1) {
68  txbuf[1] = val;
69  i2cMasterTransmitTimeout(&I2CD1, STMPE811_ADDR, txbuf, 2, NULL, 0, MS2ST(STMPE811_TIMEOUT));
70  } else if (n == 2) {
71  txbuf[1] = ((val & 0xFF00) >> 8);
72  txbuf[2] = (val & 0x00FF);
73  i2cMasterTransmitTimeout(&I2CD1, STMPE811_ADDR, txbuf, 3, NULL, 0, MS2ST(STMPE811_TIMEOUT));
74  }
75 
76  i2cReleaseBus(&I2CD1);
77 }

79 /**
80  * @brief Read the value of a certain register
81  *
82  * @param[in] reg The register address
83  * @param[in] n The amount of bytes (one or two)
84  *
85  * @return Data read from device (one byte or two depending on n param)
86  *
87  * @notapi
88  */
89 static uint16_t read_reg(uint8_t reg, uint8_t n)
90 {
91  uint8_t txbuf[1], rxbuf[2];
92  uint16_t ret;
93 
94  rxbuf[0] = 0;
95  rxbuf[1] = 0;
96 
97  i2cAcquireBus(&I2CD1);
98 
99  txbuf[0] = reg;
100  i2cMasterTransmitTimeout(&I2CD1, STMPE811_ADDR, txbuf, 1, rxbuf, n, MS2ST(STMPE811_TIMEOUT));
101 
102  if (n == 1) {
103  ret = rxbuf[0];
104  } else if (n == 2) {
105  ret = ((rxbuf[0] << 8) | (rxbuf[1] & 0xFF));
106  }
107 
108  i2cReleaseBus(&I2CD1);
109 
110  return ret;
111 }
