126 /** Flip display content horizontally.
127  * NOTE: This only flips display content, but doesn't turn the char writing around.
128  * You have to unmirror everything manually.
129  * @param[in] value 0 = column address 0 = segment 0 (the default), 1 = column address 127 = segment 0 (flipped).
130  * @notapi
131  */
132 static void flip_display(unsigned char enable) {
133  if( enable && GDISP.Orientation == GDISP_ROTATE_0) {
134  set_com_output_scan_direction(0);
135  set_segment_remap(0);
136  GDISP.Orientation = GDISP_ROTATE_0;
137  }
138  if( !enable && GDISP.Orientation == GDISP_ROTATE_180) {
139  set_com_output_scan_direction(1);
140  set_segment_remap(1);
141  GDISP.Orientation = GDISP_ROTATE_180;
142  }
143  else
144  return;
145 }

267 /**
268  * @brief Low level GDISP driver initialization.
269  *
270  * @notapi
271  */
272 bool_t gdisp_lld_init(void) {
273  // Initialize your display
274  init_board();
275 
276  // Hardware reset.
277  setpin_reset(TRUE);
278  delayms(1);
279  setpin_reset(FALSE);
280  delayms(10);
281  setpin_reset(TRUE);
282 
283  // Get the bus for the following initialization commands.
284  acquire_bus();
285 
286  display_off();
287  set_display_clock_ratio_and_frequency(0, 8);
288  #if GDISP_SCREEN_HEIGHT == 64
289  set_multiplex_ratio(0x3F); // 1/64 duty
290  #endif
291  #if GDISP_SCREEN_HEIGHT == 32
292  set_multiplex_ratio(0x1F); // 1/32 duty
293  #endif
294  set_precharge_period(0xF, 0x01); //
295  set_display_offset(0); //
296  set_display_start_line(0); //
297  set_charge_pump(1); // Enable internal charge pump.
298  set_memory_addressing_mode(0); // horizontal addressing mode; across then down //act like ks0108 (horizontal addressing mode)
299  set_segment_remap(1); //
300  set_com_output_scan_direction(1); //
301  #if GDISP_SCREEN_HEIGHT == 64
302  set_com_pins_hardware_configuration(1, 0);
303  #endif
304  #if GDISP_SCREEN_HEIGHT == 32
305  set_com_pins_hardware_configuration(0, 1);
306  #endif
307  set_contrast(GDISP_INITIAL_CONTRAST); // Set initial contrast.
308  set_vcomh_deselect_level(1); //
309  display_on(); // Turn on OLED panel.
310  invert_display(0); // Disable Inversion of display.
311  set_column_address_hvam(0, 127); //
312  set_page_address_hvam(0, 7); //
313 
314  release_bus();
315 
316  gdisp_lld_display();
317 
318  // Initialize the GDISP structure
319  GDISP.Width = GDISP_SCREEN_WIDTH;
320  GDISP.Height = GDISP_SCREEN_HEIGHT;
321  GDISP.Orientation = GDISP_ROTATE_0;
322  GDISP.Powermode = powerOn;
323  GDISP.Contrast = GDISP_INITIAL_CONTRAST;
324  #if GDISP_NEED_VALIDATION || GDISP_NEED_CLIP
325  GDISP.clipx0 = 0;
326  GDISP.clipy0 = 0;
327  GDISP.clipx1 = GDISP.Width;
328  GDISP.clipy1 = GDISP.Height;
329  #endif
330  return TRUE;
331 }

334  * @brief Draws a pixel on the display.
335  *
336  * @param[in] x X location of the pixel
337  * @param[in] y Y location of the pixel
338  * @param[in] color The color of the pixel

342 void gdisp_lld_draw_pixel(coord_t x, coord_t y, color_t color) {
343  #if GDISP_NEED_VALIDATION || GDISP_NEED_CLIP
344  if (x < GDISP.clipx0 || y < GDISP.clipy0 || x >= GDISP.clipx1 || y >= GDISP.clipy1) return;
345  #endif
346 
347  if (color == SSD1306_WHITE)
348  gdisp_buffer[x+ (y/8)*GDISP_SCREEN_WIDTH] |= (1<<y%8);
349  else
350  gdisp_buffer[x+ (y/8)*GDISP_SCREEN_WIDTH] &= ~(1<<y%8);

353 void gdisp_lld_display() {
354  set_display_start_line(0);
355 
356  /* We're sending half a line in one X-mission.*/
357  uint8_t command[GDISP_SCREEN_WIDTH/2],
358  cmdLength = sizeof(command)/sizeof(command[0]),
359  parts = GDISP_SCREEN_WIDTH/cmdLength;
360 
361  for(int i=0; i<GDISP_SCREEN_HEIGHT/8; i++){
362  for(int j = 0; j<parts; j++){
363  memmove(command, &gdisp_buffer[i*GDISP_SCREEN_WIDTH + j*cmdLength], cmdLength);
364  write_data(command, cmdLength);
365  }
366  }
367 }

387  /**
388  * @brief Scroll vertically a section of the screen.
389  * @note Optional.
390  * @note If x,y + cx,cy is off the screen, the result is undefined.
391  * @note If lines is >= cy, it is equivalent to a area fill with bgcolor.
392  *
393  * @param[in] x, y The start of the area to be scrolled
394  * @param[in] cx, cy The size of the area to be scrolled
395  * @param[in] lines The number of lines to scroll (Can be positive or negative)
396  * @param[in] bgcolor The color to fill the newly exposed area.
397  *
398  * @notapi
399  */
400  void gdisp_lld_vertical_scroll(coord_t x, coord_t y, coord_t cx, coord_t cy, int lines, color_t bgcolor) {
401  #if GDISP_NEED_VALIDATION || GDISP_NEED_CLIP
402  if (x < GDISP.clipx0) { cx -= GDISP.clipx0 - x; x = GDISP.clipx0; }
403  if (y < GDISP.clipy0) { cy -= GDISP.clipy0 - y; y = GDISP.clipy0; }
404  if (!lines || cx <= 0 || cy <= 0 || x >= GDISP.clipx1 || y >= GDISP.clipy1) return;
405  if (x+cx > GDISP.clipx1) cx = GDISP.clipx1 - x;
406  if (y+cy > GDISP.clipy1) cy = GDISP.clipy1 - y;
407  #endif
408 
409  /* See datasheet table T10-1 for this*/
410  uint8_t fHeight = (uint8_t)gdispGetFontMetric(gwinGetDefaultFont(), fontLineSpacing);
411  set_multiplex_ratio(GDISP_SCREEN_HEIGHT - fHeight+1);
412  set_display_offset(fHeight-2);
413 
414  /* Scrolling animation.*/
415  for(int i=0; i<fHeight; i++){
416  set_display_start_line(i);
417  gfxSleepMilliseconds(10);
418  }
419 
420  /* Shift buffer up a font line.*/
421  for (int i = 0; i < GDISP_SCREEN_WIDTH*(GDISP_SCREEN_HEIGHT/8-1); i++) {
422  gdisp_buffer[i] = gdisp_buffer[i+GDISP_SCREEN_WIDTH*(fHeight/8)] >> fHeight % 8;
423  gdisp_buffer[i] |= gdisp_buffer[i+GDISP_SCREEN_WIDTH*(fHeight/8 + 1)] << (8 - fHeight%8);
424  }
425 
426  /* Clear last page.*/
427  memset( &gdisp_buffer[GDISP_SCREEN_HEIGHT*GDISP_SCREEN_WIDTH/8 - GDISP_SCREEN_WIDTH*2], SSD1306_BLACK, GDISP_SCREEN_WIDTH*2);
428 
429  /* Update display.*/
430  gdisp_lld_display();
431  }

549 #if (GDISP_NEED_CONTROL && GDISP_HARDWARE_CONTROL) || defined(__DOXYGEN__)
550  /**
551  * @brief Driver Control
552  * @details Unsupported control codes are ignored.
553  * @note The value parameter should always be typecast to (void *).
554  * @note There are some predefined and some specific to the low level driver.
555  * @note GDISP_CONTROL_POWER - Takes a gdisp_powermode_t
556  * GDISP_CONTROL_ORIENTATION - Takes a gdisp_orientation_t
557  * GDISP_CONTROL_BACKLIGHT - Takes an int from 0 to 100. For a driver
558  * that only supports off/on anything other
559  * than zero is on.
560  * GDISP_CONTROL_CONTRAST - Takes an int from 0 to 100.
561  * GDISP_CONTROL_LLD - Low level driver control constants start at
562  * this value.
563  *
564  * @param[in] what What to do.
565  * @param[in] value The value to use (always cast to a void *).
566  *
567  * @notapi
568  */
569  void gdisp_lld_control(unsigned what, void *value) {
570  switch(what) {
571  case GDISP_CONTROL_POWER:
572  if (GDISP.Powermode == (gdisp_powermode_t)value)
573  return;
574  switch((gdisp_powermode_t)value) {
