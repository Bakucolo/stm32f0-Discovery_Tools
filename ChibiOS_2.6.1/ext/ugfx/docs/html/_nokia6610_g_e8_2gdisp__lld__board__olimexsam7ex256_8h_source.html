48 /**
49  * @brief Initialise the board for the display.
50  * @notes Performs the following functions:
51  * 1. initialise the spi port used by your display
52  * 2. initialise the reset pin (initial state not-in-reset)
53  * 3. initialise the chip select pin (initial state not-active)
54  * 4. initialise the backlight pin (initial state back-light off)
55  *
56  * @notapi
57  */
58 static inline void init_board(void) {
59  // *********************************************************************************************
60  // InitSpi( )
61  //
62  // Sets up SPI channel 0 for communications to Nokia 6610 LCD Display
63  //
64  // I/O ports used: PA2 = LCD Reset (set to low to reset)
65  // PA12 = LCD chip select (set to low to select the LCD chip)
66  // PA16 = SPI0_MISO Master In - Slave Out (not used in LCD interface)
67  // PA17 = SPI0_MOSI Master Out - Slave In pin (Serial Data to LCD slave)
68  // PA18 = SPI0_SPCK Serial Clock (to LCD slave)
69  // PB20 = backlight control (normally PWM control, 1 = full on)
70  //
71  // *********************************************************************************************}
72 
73  /* This code should really use the ChibiOS driver for these functions */
74 
75  // Pin for backlight
76  pPIOB->PIO_CODR = PIOB_LCD_BL_MASK; // Set PB20 to LOW
77  pPIOB->PIO_OER = PIOB_LCD_BL_MASK; // Configure PB20 as output
78 
79  // Reset pin
80  pPIOA->PIO_SODR = PIOA_LCD_RESET_MASK; // Set PA2 to HIGH
81  pPIOA->PIO_OER = PIOA_LCD_RESET_MASK; // Configure PA2 as output
82 
83  // CS pin - this seems to be ignored
84  // pPIOA->PIO_SODR = 1<<12; // Set PA2 to HIGH
85  // pPIOA->PIO_OER = 1<<12; // Configure PA2 as output
86 
87  // Init SPI0
88  // Disable the following pins from PIO control (will be used instead by the SPI0 peripheral)
89  // BIT12 = PA12 -> SPI0_NPCS0 chip select
90  // BIT16 = PA16 -> SPI0_MISO Master In - Slave Out (not used in LCD interface)
91  // BIT17 = PA17 -> SPI0_MOSI Master Out - Slave In pin (Serial Data to LCD slave)
92  // BIT18 = PA18 -> SPI0_SPCK Serial Clock (to LCD slave)
93  pPIOA->PIO_PDR = (1<<12) | (1<<16) | (1<<17) | (1<<18);
94  pPIOA->PIO_ASR = (1<<12) | (1<<16) | (1<<17) | (1<<18);
95  pPIOA->PIO_BSR = 0;
96 
97  //enable the clock of SPI
98  pPMC->PMC_PCER = 1 << AT91C_ID_SPI0;
99 
100  // Fixed mode
101  pSPI->SPI_CR = 0x81; //SPI Enable, Sowtware reset
102  pSPI->SPI_CR = 0x01; //SPI Enable
103 
104  //pSPI->SPI_MR = 0xE0019; //Master mode, fixed select, disable decoder, FDIV=1 (MCK), PCS=1110
105  pSPI->SPI_MR = 0xE0011; //Master mode, fixed select, disable decoder, FDIV=0 (MCK), PCS=1110
106 
107  //pSPI->SPI_CSR[0] = 0x01010C11; //9bit, CPOL=1, ClockPhase=0, SCLK = 48Mhz/32*12 = 125kHz
108  pSPI->SPI_CSR[0] = 0x01010311; //9bit, CPOL=1, ClockPhase=0, SCLK = 48Mhz/8 = 6MHz if using commented MR line above
109 
110  /* Display backlight control at 100% */
111  pwmRunning = FALSE;
112  palSetPad(IOPORT2, PIOB_LCD_BL);
113 }

129 /**
130  * @brief Set the lcd back-light level.
131  * @note For now 0% turns the backlight off, anything else the backlight is on.
132  * While the hardware supports PWM backlight control, we are not using it
133  * yet.
134  *
135  * @param[in] percent 0 to 100%
136  *
137  * @notapi
138  */
139 static inline void set_backlight(uint8_t percent) {
140  if (percent == 100) {
141  /* Turn the pin on - No PWM */
142  if (pwmRunning) {
143  pwmStop(&PWMD2);
144  pwmRunning = FALSE;
145  }
146  palSetPad(IOPORT2, PIOB_LCD_BL);
147  } else if (percent == 0) {
148  /* Turn the pin off - No PWM */
149  if (pwmRunning) {
150  pwmStop(&PWMD2);
151  pwmRunning = FALSE;
152  }
153  palClearPad(IOPORT2, PIOB_LCD_BL);
154  } else {
155  /* Use the PWM */
156  if (!pwmRunning) {
157  pwmStart(&PWMD2, &pwmcfg);
158  pwmRunning = TRUE;
159  }
160  pwmEnableChannel(&PWMD2, 0, PWM_VALUE(percent));
161  }
162 }
