58 static void draw_point(coord_t x, coord_t y) {
59  color_t c = ncoreDrawingArea->color;
60 
61  if (nMode == NCORE_MODE_DRAW)
62  c = ncoreDrawingArea->color;

64  c = ncoreDrawingArea->bgcolor;
65 
66  if (nPenWidth == 1)
67  gdispDrawPixel(x, y, c);
68  else

72 /* Bresenham's Line Drawing Algorithm
73  Modified version to draw line of variable thickness */
74 static void draw_line(coord_t x0, coord_t y0, coord_t x1, coord_t y1) {
75  int16_t dy, dx;
76  int16_t addx, addy;
77  int16_t P, diff, i;
78 
79  if (x1 >= x0) {
80  dx = x1 - x0;
81  addx = 1;
82  } else {
83  dx = x0 - x1;
84  addx = -1;
85  }
86  if (y1 >= y0) {
87  dy = y1 - y0;
88  addy = 1;
89  } else {
90  dy = y0 - y1;
91  addy = -1;
92  }
93 
94  if (dx >= dy) {
95  dy *= 2;
96  P = dy - dx;
97  diff = P - dx;
98 
99  for(i=0; i<=dx; ++i) {
100  draw_point(x0, y0);
101  if (P < 0) {
102  P += dy;
103  x0 += addx;
104  } else {
105  P += diff;
106  x0 += addx;
107  y0 += addy;
108  }
109  }
110  } else {
111  dx *= 2;
112  P = dx - dy;
113  diff = P - dy;
114 
115  for(i=0; i<=dy; ++i) {
116  draw_point(x0, y0);
117  if (P < 0) {
118  P += dx;
119  y0 += addy;
120  } else {
121  P += diff;
122  x0 += addx;
123  y0 += addy;
124  }
125  }
126  }
127 }

129 /* Core thread */
130 static DECLARE_THREAD_FUNCTION(ncoreDrawThread, msg) {
131 
132  GEventMouse ev, evPrev;
133  coord_t dx, dy;
134 
135  int state = 0, dist;
136 
137  (void)msg;
138 
139  ginputGetMouseStatus(0, &evPrev);
140 
141  while (!doExit) {
142 
143  ginputGetMouseStatus(0, &ev);
144  switch(state) {
145  case 0: if (ev.meta == GMETA_MOUSE_DOWN) {
146  state = 1;
147  if (nMode == NCORE_MODE_FILL && PEN_IN_DRAWING_AREA(ev)) {
148  // Set bgcolor to current color, clear the display.
149  ncoreDrawingArea->bgcolor = ncoreDrawingArea->color;
150  gwinClear(ncoreDrawingArea);
151  }
152  }
153  else
154  gfxYield();
155  break;
156 
157 
158  case 1: if (ev.meta == GMETA_MOUSE_UP) {
159  state = 0;
160  //gwinPrintf(nStatusConsole, "\nPen Up: (%d, %d)", ev.x, ev.y);
161  break;
162  }
163 
164  dx = abs(ev.x - evPrev.x);
165  dy = abs(ev.y - evPrev.y);
166 
167  dist = dx * dx + dy * dy;
168 
169  if (dist > 0)
170  {
171  gdispSetClip(ncoreDrawingArea->x,
172  ncoreDrawingArea->y,
173  ncoreDrawingArea->width,
174  ncoreDrawingArea->height);
175 
176  if (PEN_IN_DRAWING_AREA(ev)){
177  // Do Interpolation
178  if (dist <= 2) {
179  draw_point(ev.x, ev.y);
180  }
181  else if (dist <= 5) {
182  // Line drawing does not give good results for this case.
183  // So draw two pixels directly
184  draw_point(ev.x, ev.y);
185  draw_point((ev.x + evPrev.x) / 2, (ev.y + evPrev.y) / 2);
186  }
187  else if (dx * dx <= MAX_DX && dy * dy <= MAX_DY) {
188  draw_line(ev.x, ev.y, evPrev.x, evPrev.y);
189  }
190  }
191 
192  //gwinPrintf(nStatusConsole, "\nPen Down: (%d, %d)", ev.x, ev.y);
193  }
194  break;
195  }
196  evPrev = ev;
197  }
198 
199  return 0;
200 }
