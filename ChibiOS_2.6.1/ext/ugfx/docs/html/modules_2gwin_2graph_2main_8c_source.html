63 int main(void) {
64  GHandle gh;
65  uint16_t i;
66 
67  gfxInit();
68 
69  {
70  GWindowInit wi;
71 
72  wi.show = TRUE;
73  wi.x = wi.y = 0;
74  wi.width = gdispGetWidth();
75  wi.height = gdispGetHeight();
76  gh = gwinGraphCreate(&g, &wi);
77  }
78 
79  gwinGraphSetOrigin(gh, gwinGetWidth(gh)/2, gwinGetHeight(gh)/2);
80  gwinGraphSetStyle(gh, &GraphStyle1);
81  gwinGraphDrawAxis(gh);
82 
83  for(i = 0; i < gwinGetWidth(gh); i++)
84  gwinGraphDrawPoint(gh, i-gwinGetWidth(gh)/2, 80*fsin(2*i/5)); //sin(2*0.2*M_PI*i/180));
85 
86  gwinGraphStartSet(gh);
87  GraphStyle1.point.color = Green;
88  gwinGraphSetStyle(gh, &GraphStyle1);
89 
90  for(i = 0; i < gwinGetWidth(gh)*5; i++)
91  gwinGraphDrawPoint(gh, i/5-gwinGetWidth(gh)/2, 95*fsin(2*i/5)); //sin(2*0.2*M_PI*i/180));
92 
93  gwinGraphStartSet(gh);
94  gwinGraphSetStyle(gh, &GraphStyle2);
95 
96  gwinGraphDrawPoints(gh, data, sizeof(data)/sizeof(data[0]));
97 
98  while(TRUE) {
99  gfxSleepMilliseconds(100);
100  }
101 }
