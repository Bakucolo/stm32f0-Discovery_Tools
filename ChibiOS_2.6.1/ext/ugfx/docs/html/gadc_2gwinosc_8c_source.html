47 static void _destroy(GHandle gh) {
48  gadcHighSpeedStop();

51  ((GScopeObject *)gh)->lastscopetrace = 0;
52  }
53  if (((GScopeObject *)gh)->audiobuf) {
54  gfxFree(((GScopeObject *)gh)->audiobuf);
55  ((GScopeObject *)gh)->audiobuf = 0;

67 GHandle gwinScopeCreate(GScopeObject *gs, GWindowInit *pInit, uint32_t physdev, uint32_t frequency) {
68  /* Initialise the base class GWIN */
69  if (!(gs = (GScopeObject *)_gwindowCreate(&gs->g, pInit, &scopeVMT, 0)))
70  return 0;
71  gfxSemInit(&gs->bsem, 0, 1);
72  gs->nextx = 0;
73  if (!(gs->lastscopetrace = (coord_t *)gfxAlloc(gs->g.width * sizeof(coord_t))))
74  return 0;
75  if (!(gs->audiobuf = (adcsample_t *)gfxAlloc(AUDIOBUFSZ * sizeof(adcsample_t))))
76  return 0;
77 #if TRIGGER_METHOD == TRIGGER_POSITIVERAMP
78  gs->lasty = gs->g.height/2;
79 #elif TRIGGER_METHOD == TRIGGER_MINVALUE
80  gs->lasty = gs->g.height/2;
81  gs->scopemin = 0;
82 #endif
83 
84  /* Start the GADC high speed converter */
85  gadcHighSpeedInit(physdev, frequency, gs->audiobuf, AUDIOBUFSZ, AUDIOBUFSZ/2);
86  gadcHighSpeedSetBSem(&gs->bsem, &gs->myEvent);
87  gadcHighSpeedStart();
88 
89  gwinSetVisible((GHandle)gs, pInit->show);
90  return (GHandle)gs;
91 }

93 void gwinScopeWaitForTrace(GHandle gh) {
94  #define gs ((GScopeObject *)(gh))
95  int i;
96  coord_t x, y;
97  coord_t yoffset;
98  adcsample_t *pa;
99  coord_t *pc;
100 #if TRIGGER_METHOD == TRIGGER_POSITIVERAMP
101  bool_t rdytrigger;
102  int flsamples;
103 #elif TRIGGER_METHOD == TRIGGER_MINVALUE
104  bool_t rdytrigger;
105  int flsamples;
106  coord_t scopemin;
107 #endif
108 
109  if (gh->vmt != &scopeVMT)
110  return;
111 
112  /* Wait for a set of audio conversions */
113  gfxSemWait(&gs->bsem, TIME_INFINITE);
114 
115  /* Ensure we are drawing in the right area */
116  #if GDISP_NEED_CLIP
117  gdispSetClip(gh->x, gh->y, gh->width, gh->height);
118  #endif
119 
120  yoffset = gh->height/2 + (1<<SCOPE_Y_BITS)/2;
121  x = gs->nextx;
122  pc = gs->lastscopetrace+x;
123  pa = gs->myEvent.buffer;
124 #if TRIGGER_METHOD == TRIGGER_POSITIVERAMP
125  rdytrigger = FALSE;
126  flsamples = 0;
127 #elif TRIGGER_METHOD == TRIGGER_MINVALUE
128  rdytrigger = FALSE;
129  flsamples = 0;
130  scopemin = 0;
131 #endif
132 
133  for(i = gs->myEvent.count; i; i--) {
134 
135  /* Calculate the new scope value - re-scale using simple shifts for efficiency, re-center and y-invert */
136  #if GADC_BITS_PER_SAMPLE > SCOPE_Y_BITS
137  y = yoffset - (*pa++ >> (GADC_BITS_PER_SAMPLE - SCOPE_Y_BITS));
138  #else
139  y = yoffset - (*pa++ << (SCOPE_Y_BITS - GADC_BITS_PER_SAMPLE));
140  #endif
141 
142 #if TRIGGER_METHOD == TRIGGER_MINVALUE
143  /* Calculate the scopemin ready for the next trace */
144  if (y > scopemin)
145  scopemin = y;
146 #endif
147 
148  /* Have we reached the end of a scope trace? */
149  if (x >= gh->width) {
150 
151 #if TRIGGER_METHOD == TRIGGER_POSITIVERAMP || TRIGGER_METHOD == TRIGGER_MINVALUE
152  /* Handle triggering - we trigger on the next sample minimum (y value maximum) or a flat-line */
153 
154  #if TRIGGER_METHOD == TRIGGER_MINVALUE
155  /* Arm when we reach the sample minimum (y value maximum) of the previous trace */
156  if (!rdytrigger && y >= gs->scopemin)
157  rdytrigger = TRUE;
158  #endif
159 
160  if (y == gs->lasty) {
161  /* Trigger if we get too many flat-line samples regardless of the armed state */
162  if (++flsamples < FLATLINE_SAMPLES)
163  continue;
164  flsamples = 0;
165  } else if (y > gs->lasty) {
166  gs->lasty = y;
167  flsamples = 0;
168  #if TRIGGER_METHOD == TRIGGER_POSITIVERAMP
169  /* Arm the trigger when samples fall (y increases) ie. negative slope */
170  rdytrigger = TRUE;
171  #endif
172  continue;
173  } else {
174  /* If the trigger is armed, Trigger when samples increases (y decreases) ie. positive slope */
175  gs->lasty = y;
176  flsamples = 0;
177  if (!rdytrigger)
178  continue;
179  }
180 
181  /* Ready for a the next trigger cycle */
182  rdytrigger = FALSE;
183 #endif
184 
185  /* Prepare for a scope trace */
186  x = 0;
187  pc = gs->lastscopetrace;
188  }
189 
190  /* Clear the old scope pixel and then draw the new scope value */
191  gdispDrawPixel(gh->x+x, gh->y+pc[0], gh->bgcolor);
192  gdispDrawPixel(gh->x+x, gh->y+y, gh->color);
193 
194  /* Save the value */
195  *pc++ = y;
196  x++;
197  #if TRIGGER_METHOD == TRIGGER_POSITIVERAMP || TRIGGER_METHOD == TRIGGER_MINVALUE
198  gs->lasty = y;
199  #endif
200  }
201  gs->nextx = x;
202 #if TRIGGER_METHOD == TRIGGER_MINVALUE
203  gs->scopemin = scopemin;
204 #endif
205 
206  #undef gs
207 }
