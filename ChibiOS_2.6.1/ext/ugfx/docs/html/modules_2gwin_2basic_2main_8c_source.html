32 /* The handles for our two Windows */
33 GHandle GW1, GW2;
34 
35 int main(void) {
36  coord_t i, j;
37 
38  /* Initialize and clear the display */
39  gfxInit();
40  gdispClear(White);
41 
42  /* Create two windows */
43  {
44  GWindowInit wi;
45 
46  wi.show = TRUE; wi.x = 20; wi.y = 10; wi.width = 200; wi.height = 150;
47  GW1 = gwinWindowCreate(NULL, &wi);
48  wi.show = TRUE; wi.x = 50; wi.y = 190; wi.width = 150; wi.height = 100;
49  GW2 = gwinWindowCreate(NULL, &wi);
50  }
51 
52  /* Set fore- and background colors for both windows */
53  gwinSetColor(GW1, Black);
54  gwinSetBgColor(GW1, White);
55  gwinSetColor(GW2, White);
56  gwinSetBgColor(GW2, Blue);
57 
58  /* Clear both windows - to set background color */
59  gwinClear(GW1);
60  gwinClear(GW2);
61 
62  gwinDrawLine(GW1, 5, 30, 150, 110);
63  for(i = 5, j = 0; i < 200 && j < 150; i += 3, j += i/20)
64  gwinDrawPixel(GW1, i, j);
65 
66  /*
67  * Draw two filled circles at the same coordinate
68  * of each window to demonstrate the relative coordinates
69  * of windows
70  */
71  gwinFillCircle(GW1, 20, 20, 15);
72  gwinFillCircle(GW2, 20, 20, 15);
73 
74  while(TRUE) {
75  gfxSleepMilliseconds(500);
76  }
77 }
