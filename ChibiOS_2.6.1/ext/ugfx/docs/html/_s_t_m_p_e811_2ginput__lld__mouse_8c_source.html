46 /**
47  * @brief Initialise the mouse/touch.
48  *
49  * @notapi
50  */
51 void ginput_lld_mouse_init(void)
52 {
53  init_board();
54 
55  write_reg(STMPE811_REG_SYS_CTRL1, 1, 0x02); // Software chip reset
56  gfxSleepMilliseconds(10);
57 
58  write_reg(STMPE811_REG_SYS_CTRL2, 1, 0x0C); // Temperature sensor clock off, GPIO clock off, touch clock on, ADC clock on
59 #if STMP811_NO_GPIO_IRQPIN
60  write_reg(STMPE811_REG_INT_EN, 1, 0x00); // Interrupt on INT pin when touch is detected
61 #else
62  write_reg(STMPE811_REG_INT_EN, 1, 0x01); // Interrupt on INT pin when touch is detected
63 #endif
64  write_reg(STMPE811_REG_ADC_CTRL1, 1, 0x48); // ADC conversion time = 80 clock ticks, 12-bit ADC, internal voltage refernce
65  gfxSleepMilliseconds(2);
66 
67  write_reg(STMPE811_REG_ADC_CTRL2, 1, 0x01); // ADC speed 3.25MHz
68  write_reg(STMPE811_REG_GPIO_AF, 1, 0x00); // GPIO alternate function - OFF
69  write_reg(STMPE811_REG_TSC_CFG, 1, 0x9A); // Averaging 4, touch detect delay 500 us, panel driver settling time 500 us
70  write_reg(STMPE811_REG_FIFO_TH, 1, 0x40); // FIFO threshold = 64
71  write_reg(STMPE811_REG_FIFO_STA, 1, 0x01); // FIFO reset enable
72  write_reg(STMPE811_REG_FIFO_STA, 1, 0x00); // FIFO reset disable
73  write_reg(STMPE811_REG_TSC_FRACT_XYZ, 1, 0x07); // Z axis data format
74  write_reg(STMPE811_REG_TSC_I_DRIVE, 1, 0x01); // 50mA touchscreen line current
75  write_reg(STMPE811_REG_TSC_CTRL, 1, 0x00); // X&Y&Z
76  write_reg(STMPE811_REG_TSC_CTRL, 1, 0x01); // X&Y&Z, TSC enable
77  write_reg(STMPE811_REG_INT_STA, 1, 0xFF); // Clear all interrupts
78 #if !STMP811_NO_GPIO_IRQPIN
79  touched = (uint8_t)read_reg(STMPE811_REG_TSC_CTRL, 1) & 0x80;
80 #endif
81  write_reg(STMPE811_REG_INT_CTRL, 1, 0x01); // Level interrupt, enable intrrupts
82 }

84 /**
85  * @brief Read the mouse/touch position.
86  *
87  * @param[in] pt A pointer to the structure to fill
88  *
89  * @note For drivers that don't support returning a position
90  * when the touch is up (most touch devices), it should
91  * return the previous position with the new Z value.
92  * The z value is the pressure for those touch devices
93  * that support it (-100 to 100 where > 0 is touched)
94  * or, 0 or 100 for those drivers that don't.
95  *
96  * @notapi
97  */
98 void ginput_lld_mouse_get_reading(MouseReading *pt)
99 {
100  bool_t clearfifo; // Do we need to clear the FIFO
101 
102 #if STMP811_NO_GPIO_IRQPIN
103  // Poll to get the touched status
104  uint8_t last_touched;
105 
106  last_touched = touched;
107  touched = (uint8_t)read_reg(STMPE811_REG_TSC_CTRL, 1) & 0x80;
108  clearfifo = (touched != last_touched);
109 #else
110  // Check if the touch controller IRQ pin has gone off
111  clearfifo = false;
112  if(getpin_irq()) { // please rename this to getpin_irq
113  write_reg(STMPE811_REG_INT_STA, 1, 0xFF); // clear all interrupts
114  touched = (uint8_t)read_reg(STMPE811_REG_TSC_CTRL, 1) & 0x80; // set the new touched status
115  clearfifo = true; // only take the last FIFO reading
116  }
117 #endif
118 
119  // If not touched, return the previous results
120  if (!touched) {
121  pt->x = x;
122  pt->y = y;
123  pt->z = 0;
124  pt->buttons = 0;
125  return;
126  }
127 
128 #if !STMP811_SLOW_CPU
129  if (!clearfifo && (read_reg(STMPE811_REG_FIFO_STA, 1) & 0xD0))
130 #endif
131  clearfifo = true;
132 
133  do {
134  /* Get the X, Y, Z values */
135  /* This could be done in a single 4 byte read to STMPE811_REG_TSC_DATA_XYZ (incr or non-incr) */
136  x = (coord_t)read_reg(STMPE811_REG_TSC_DATA_X, 2);
137  y = (coord_t)read_reg(STMPE811_REG_TSC_DATA_Y, 2);
138  z = (coord_t)read_reg(STMPE811_REG_TSC_DATA_Z, 1);
139  } while(clearfifo && !(read_reg(STMPE811_REG_FIFO_STA, 1) & 0x20));
140 
141  // Rescale X,Y,Z - X & Y don't need scaling when you are using calibration!
142 #if !GINPUT_MOUSE_NEED_CALIBRATION
143  x = gdispGetWidth() - x / (4096/gdispGetWidth());
144  y = y / (4096/gdispGetHeight());
145 #endif
146  z = (((z&0xFF) * 100)>>8) + 1;
147 
148  // Return the results. ADC gives values from 0 to 2^12 (4096)
149  pt->x = x;
150  pt->y = y;
151  pt->z = z;
152  pt->buttons = GINPUT_TOUCH_PRESSED;
153 
154  /* Force another read if we have more results */
155  if (!clearfifo && !(read_reg(STMPE811_REG_FIFO_STA, 1) & 0x20))
156  ginputMouseWakeup();
157 
158 }
159 
160 #endif /* GFX_USE_GINPUT && GINPUT_NEED_MOUSE */
161 /** @} */
