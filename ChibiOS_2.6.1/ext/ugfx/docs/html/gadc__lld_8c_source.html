37 size_t gadc_lld_samples_per_conversion(uint32_t physdev) {
38  size_t cnt;
39  int i;
40 
41  /* The AT91SAM7 has AD0..7 - physdev is a bitmap of those channels */
42  for(cnt = 0, i = 0; i < 8; i++, physdev >>= 1)
43  if (physdev & 0x01)
44  cnt++;
45  return cnt;
46 }

48 void gadc_lld_start_timer(uint32_t physdev, uint32_t frequency) {
49  (void) physdev;
50  /**
51  * The AT91SAM7 ADC driver supports triggering the ADC using a timer without having to implement
52  * an interrupt handler for the timer. The driver also initialises the timer correctly for us.
53  * Because we aren't trapping the interrupt ourselves we can't increment GADC_Timer_Missed if an
54  * interrupt is missed.
55  */
56  acg.frequency = frequency;
57 }

59 void gadc_lld_stop_timer(uint32_t physdev) {
60  (void) physdev;
61  if ((acg.trigger & ~ADC_TRIGGER_SOFTWARE) == ADC_TRIGGER_TIMER)
62  adcStop(&ADCD1);
63 }

65 void gadc_lld_adc_timerI(GadcLldTimerData *pgtd) {
66  /**
67  * We don't need to calculate num_channels because the AT91SAM7 ADC does this for us.
68  */
69  acg.channelselects = pgtd->physdev;
70  acg.trigger = pgtd->now ? (ADC_TRIGGER_TIMER|ADC_TRIGGER_SOFTWARE) : ADC_TRIGGER_TIMER;
71 
72  adcStartConversionI(&ADCD1, &acg, pgtd->buffer, pgtd->count);
73 
74  /* Next time assume the same (still running) timer */
75  acg.frequency = 0;
76 }

78 void gadc_lld_adc_nontimerI(GadcLldNonTimerData *pgntd) {
79  /**

81  */
82  acg.channelselects = pgntd->physdev;
